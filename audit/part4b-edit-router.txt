    44	/** Extract symbol name from a chunk's raw code. */
    45	function extractName(chunk: ParsedChunk): string {
    46	    const raw = chunk.rawCode.trim();
    47	    let m: RegExpExecArray | null;
    48	
    49	    // TS/JS: Arrow functions (export const foo = async () => ...)
    50	    m = /(?:export\s+)?(?:default\s+)?(?:const|let|var)\s+(\w+)\s*=\s*(?:async\s+)?(?:(?:<[^>]*>\s*)?\([^)]*\)|[a-zA-Z0-9_]+)\s*=>/.exec(raw);
    51	    if (m) return m[1];
    52	
    53	    // export [default] [async] function NAME
    54	    m = /(?:export\s+)?(?:default\s+)?(?:async\s+)?function\s+(\w+)/.exec(raw);
    55	    if (m) return m[1];
    56	
    57	    // class NAME
    58	    m = /(?:export\s+)?(?:default\s+)?(?:abstract\s+)?class\s+(\w+)/.exec(raw);
    59	    if (m) return m[1];
    60	
    61	    // interface NAME
    62	    m = /(?:export\s+)?interface\s+(\w+)/.exec(raw);
    63	    if (m) return m[1];
    64	
    65	    // type NAME
    66	    m = /(?:export\s+)?type\s+(\w+)/.exec(raw);
    67	    if (m) return m[1];
    68	
    69	    // enum NAME
    70	    m = /(?:export\s+)?enum\s+(\w+)/.exec(raw);
    71	    if (m) return m[1];
    72	
    73	    // method: NAME(
    74	    m = /(?:async\s+)?(?:static\s+)?(?:readonly\s+)?(?:public\s+|private\s+|protected\s+)?(\w+)\s*[(<]/.exec(raw);
    75	    if (m) return m[1];
    76	
    77	    // Python: def NAME / class NAME
    78	    m = /(?:async\s+)?def\s+(\w+)/.exec(raw);
    79	    if (m) return m[1];
    80	    m = /class\s+(\w+)/.exec(raw);
    81	    if (m) return m[1];
    82	
    83	    // Go: func NAME / func (receiver) NAME
    84	    m = /func\s+(?:\([^)]*\)\s+)?(\w+)/.exec(raw);
    85	    if (m) return m[1];
    86	    m = /type\s+(\w+)/.exec(raw);
    87	    if (m) return m[1];
    88	
    89	    // Fallback
    90	    m = /(\w+)/.exec(raw);
    91	    return m ? m[1] : "";
    92	}

   122	/**
   123	 * Find a chunk by symbol name. Prefers AST symbolName, falls back to regex extraction.
   124	 */
   125	export function findChunkBySymbol(
   126	    chunks: ParsedChunk[],
   127	    symbolName: string,
   128	): ParsedChunk | null {
   129	    // Exact match on AST symbolName first
   130	    const astMatch = chunks.find(c => c.symbolName && c.symbolName === symbolName);
   131	    if (astMatch) return astMatch;
   132	
   133	    // Fallback: regex extraction
   134	    const regexMatch = chunks.find(c => extractName(c) === symbolName);
   135	    if (regexMatch) return regexMatch;
   136	
   137	    // Case-insensitive fallback
   138	    const lower = symbolName.toLowerCase();
   139	    const fuzzy = chunks.find(c => {
   140	        const name = c.symbolName || extractName(c);
   141	        return name.toLowerCase() === lower;
   142	    });
   143	    return fuzzy ?? null;
   144	}

   146	/**
   147	 * Apply a semantic splice on a string in RAM. Pure function: string in → string out.
   148	 * Verifies byte indices, rebases indentation, and splices by mode.
   149	 */
   150	export function applySemanticSplice(
   151	    content: string,
   152	    target: SpliceTarget,
   153	    newCode: string,
   154	    mode: EditMode = "replace",
   155	): string {
   156	    const { startIndex, rawCode } = target;
   157	
   158	    // Verify tree-sitter byte position against actual content
   159	    let startIdx = startIndex;
   160	    if (content.substring(startIdx, startIdx + rawCode.length) !== rawCode) {
   161	        // ±500 byte search window. Duplicate symbols within range are detected and rejected.
   162	        const windowStart = Math.max(0, startIdx - 500);
   163	        const windowEnd = Math.min(content.length, startIdx + rawCode.length + 500);
   164	        const searchWindow = content.substring(windowStart, windowEnd);
   165	        let matchCount = 0;
   166	        let bestOffset = -1;
   167	        let scanPos = searchWindow.indexOf(rawCode);
   168	        while (scanPos >= 0) {
   169	            matchCount++;
   170	            if (matchCount === 1) bestOffset = scanPos;
   171	            if (matchCount > 1) break;
   172	            scanPos = searchWindow.indexOf(rawCode, scanPos + 1);
   173	        }
   174	
   175	        // AUDIT FIX: Abort if multiple identical symbols exist in the search window
   176	        if (matchCount > 1) {
   177	            throw new Error(
   178	                `Ambiguous AST match: ${matchCount} occurrences of "${target.symbolName}" found within ±500 bytes. ` +
   179	                `Use nreki_navigate action:"outline" to identify exact lines and use batch_edit.`
   180	            );
   181	        }
   182	
   183	        if (bestOffset >= 0) {
   184	            startIdx = windowStart + bestOffset;
   185	        } else {
   186	            startIdx = -1;
   187	        }
   188	
   189	        if (startIdx < 0) {
   190	            throw new Error(
   191	                `Cannot locate symbol "${target.symbolName}" in file content. AST offset mismatch.`,
   192	            );
   193	        }
   194	    }
   195	    const endIdx = startIdx + rawCode.length;
   196	
   197	    // Extract exact original indentation
   198	    let lineStart = startIdx;
   199	    while (lineStart > 0 && content[lineStart - 1] !== "\n") lineStart--;
   200	    if (content[lineStart] === "\r") lineStart++;
   201	    const indentMatch = content.slice(lineStart, startIdx).match(/^[ \t]*/);
   202	    const baseIndent = indentMatch ? indentMatch[0] : "";
   203	
   204	    // Calculate minimum indentation in the new code (SKIP first line)
   205	    const newLines = newCode.split("\n");
   206	    const nonBlankInterior = newLines.filter((l, i) => i > 0 && l.trim().length > 0);
   207	    let minClaudeIndent = Infinity;
   208	    for (const line of nonBlankInterior) {
   209	        const match = line.match(/^[ \t]*/);
   210	        if (match && match[0].length < minClaudeIndent) minClaudeIndent = match[0].length;
   211	    }
   212	    if (minClaudeIndent === Infinity) minClaudeIndent = 0;
   213	
   214	    // Relative rebase: strip Claude's indent, apply baseIndent
   215	    // First line is excluded from minClaudeIndent calc, so don't strip it
   216	    const formattedNewCode = newLines.map((line, i) => {
   217	        if (line.trim() === "") return "";
   218	        if (mode === "replace" && i === 0) return line.trimStart();
   219	
   220	        // SAFE SLICE: never cut into actual text content.
   221	        // If a line has fewer leading spaces than minClaudeIndent
   222	        // (e.g. inside template strings, multiline comments),
   223	        // only strip its actual whitespace prefix.
   224	        const actualIndentMatch = line.match(/^[ \t]*/);
   225	        const actualIndentLength = actualIndentMatch ? actualIndentMatch[0].length : 0;
   226	        const safeSliceLength = Math.min(minClaudeIndent, actualIndentLength);
   227	
   228	        const strippedLine = line.slice(safeSliceLength);
   229	        return baseIndent + strippedLine;
   230	    }).join("\n");
   231	
   232	    // Splice by mode
   233	    if (mode === "insert_before") {
   234	        return content.slice(0, lineStart) + formattedNewCode + "\n\n" + content.slice(lineStart);
   235	    } else if (mode === "insert_after") {
   236	        let endOfLine = endIdx;
   237	        while (endOfLine < content.length && content[endOfLine] !== "\n") endOfLine++;
   238	        return content.slice(0, endOfLine) + "\n\n" + formattedNewCode + content.slice(endOfLine);
   239	    } else {
   240	        return content.slice(0, startIdx) + formattedNewCode + content.slice(endIdx);
   241	    }
   242	}

   274	/**
   275	 * Apply multiple edits atomically across multiple files.
   276	 * All-or-nothing: if ANY validation fails, ZERO files are written.
   277	 */
   278	export async function batchSemanticEdit(
   279	    edits: BatchEditOp[],
   280	    parser: ASTParser,
   281	    sandbox: AstSandbox,
   282	    projectRoot: string,
   283	    dryRun: boolean = false,
   284	): Promise<BatchEditResult> {
   285	    const { safePath } = await import("./utils/path-jail.js");
   286	
   287	    if (!edits || edits.length === 0) {
   288	        return { success: false, editCount: 0, fileCount: 0, files: [], error: "No edits provided." };
   289	    }
   290	
   291	    // 1. Group edits by file, resolving paths
   292	    const editsByFile = new Map<string, BatchEditOp[]>();
   293	    for (const edit of edits) {
   294	        const resolved = safePath(projectRoot, edit.path);
   295	        const arr = editsByFile.get(resolved) || [];
   296	        arr.push(edit);
   297	        editsByFile.set(resolved, arr);
   298	    }
   299	
   300	    // 2. Build VFS in RAM
   301	    const vfs = new Map<string, string>();
   302	    for (const filePath of editsByFile.keys()) {
   303	        try {
   304	            vfs.set(filePath, readSource(filePath));
   305	        } catch (err) {
   306	            return {
   307	                success: false, editCount: edits.length, fileCount: editsByFile.size, files: [],
   308	                error: `Cannot read file "${filePath}": ${(err as Error).message}`,
   309	            };
   310	        }
   311	    }
   312	
   313	    // Track old raw codes for blast radius detection
   314	    const oldRawCodes = new Map<string, string>();
   315	
   316	    // 3. For each file: parse ONCE, map edits, reverse splice
   317	    for (const [filePath, fileEdits] of editsByFile.entries()) {
   318	        let virtualCode = vfs.get(filePath)!;
   319	        const parseResult = await parser.parse(filePath, virtualCode);
   320	
   321	        // Map each edit to its AST chunk
   322	        const mappedEdits: Array<{ edit: BatchEditOp; chunk: ParsedChunk }> = [];
   323	        for (const edit of fileEdits) {
   324	            const chunk = findChunkBySymbol(parseResult.chunks, edit.symbol);
   325	            if (!chunk) {
   326	                const available = parseResult.chunks
   327	                    .map(c => c.symbolName || extractName(c))
   328	                    .filter(Boolean)
   329	                    .join(", ");
   330	                return {
   331	                    success: false, editCount: edits.length, fileCount: editsByFile.size, files: [],
   332	                    error: `Symbol "${edit.symbol}" not found in ${edit.path}. Available: ${available}`,
   333	                };
   334	            }
   335	            mappedEdits.push({ edit, chunk });
   336	        }
   337	
   338	        // Detect overlapping ranges
   339	        for (let i = 0; i < mappedEdits.length; i++) {
   340	            for (let j = i + 1; j < mappedEdits.length; j++) {
   341	                const a = mappedEdits[i].chunk;
   342	                const b = mappedEdits[j].chunk;
   343	                if (a.startIndex < b.endIndex && b.startIndex < a.endIndex) {
   344	                    return {
   345	                        success: false, editCount: edits.length, fileCount: editsByFile.size, files: [],
   346	                        error: `Overlapping edits: "${mappedEdits[i].edit.symbol}" and "${mappedEdits[j].edit.symbol}" ` +
   347	                            `overlap in ${path.relative(projectRoot, filePath)}. Separate them into two calls.`,
   348	                    };
   349	                }
   350	            }
   351	        }
   352	
   353	        // Reverse splice: sort by startIndex DESCENDING (bottom-up)
   354	        mappedEdits.sort((a, b) => b.chunk.startIndex - a.chunk.startIndex);
   355	
   356	        for (const { edit, chunk } of mappedEdits) {
   357	            oldRawCodes.set(`${edit.path}::${edit.symbol}`, chunk.rawCode);
   358	            try {
   359	                virtualCode = applySemanticSplice(
   360	                    virtualCode,
   361	                    {
   362	                        startIndex: chunk.startIndex,
   363	                        endIndex: chunk.endIndex,
   364	                        rawCode: chunk.rawCode,
   365	                        symbolName: chunk.symbolName || extractName(chunk),
   366	                        startLine: chunk.startLine,
   367	                    },
   368	                    edit.new_code,
   369	                    edit.mode || "replace",
   370	                );
   371	            } catch (err) {
   372	                return {
   373	                    success: false, editCount: edits.length, fileCount: editsByFile.size, files: [],
   374	                    error: `Error splicing "${edit.symbol}" in ${edit.path}: ${(err as Error).message}. No files modified.`,
   375	                };
   376	            }
   377	        }
   378	
   379	        vfs.set(filePath, virtualCode);
   380	    }
   381	
   382	    // 4. Validate ALL virtual files
   383	    await sandbox.initialize();
   384	    for (const [filePath, virtualContent] of vfs.entries()) {
   385	        const language = detectLanguage(filePath);
   386	        if (!language) continue;
   387	
   388	        const validation = await sandbox.validateCode(virtualContent, language);
   389	        if (!validation.valid) {
   390	            const errDetail = validation.errors.slice(0, 3).map(e =>
   391	                `  L${e.line}:${e.column} - ${e.context.split("\n")[0].trim()}`
   392	            ).join("\n");
   393	
   394	            return {
   395	                success: false, editCount: edits.length, fileCount: editsByFile.size, files: [],
   396	                error: `TRANSACTION ABORTED - syntax error in ${path.relative(projectRoot, filePath)}:\n${errDetail}\n\n` +
   397	                    `No files were modified.\n${validation.suggestion}\nFix the code and resend the full batch.`,
   398	            };
   399	        }
   400	    }
   401	
   402	    // 5. COMMIT: Two-Phase Atomic Write with Unbreakable Rollback
   403	    const writtenFiles: string[] = [];
   404	    if (!dryRun) {
   405	        const tmpFiles = new Map<string, string>();
   406	        const renamedFiles: string[] = [];
   407	
   408	        try {
   409	            // PHASE 1: PREPARE — Mandatory backups + entropy-named temps
   410	            for (const [filePath, virtualContent] of vfs.entries()) {
   411	                const entropy = crypto.randomBytes(4).toString("hex");
   412	                const tmpPath = `${filePath}.nreki-${Date.now()}-${entropy}.tmp`;
   413	
   414	                // Backup is MANDATORY for rollback safety.
   415	                // If backup fails (disk full), abort BEFORE any rename.
   416	                if (fs.existsSync(filePath)) {
   417	                    try {
   418	                        saveBackup(projectRoot, filePath);
   419	                    } catch (backupErr) {
   420	                        for (const tp of tmpFiles.values()) {
   421	                            if (fs.existsSync(tp)) try { fs.unlinkSync(tp); } catch {}
   422	                        }
   423	                        throw new Error(`Backup failed for ${filePath}: ${(backupErr as Error).message}`);
   424	                    }
   425	                }
   426	
   427	                fs.writeFileSync(tmpPath, virtualContent, "utf-8");
   428	                tmpFiles.set(filePath, tmpPath);
   429	            }
   430	
   431	            // PHASE 2: COMMIT — Atomic POSIX inode rename
   432	            for (const [filePath, tmpPath] of tmpFiles.entries()) {
   433	                fs.renameSync(tmpPath, filePath);
   434	                renamedFiles.push(filePath);
   435	                writtenFiles.push(path.relative(projectRoot, filePath));
   436	            }
   437	        } catch (fatalErr) {
   438	            // PHASE 3: ROLLBACK — Revert successful renames from backup
   439	            for (const donePath of renamedFiles) {
   440	                const backupPath = getBackupPath(projectRoot, donePath);
   441	                if (fs.existsSync(backupPath)) {
   442	                    try { fs.copyFileSync(backupPath, donePath); } catch {}
   443	                }
   444	            }
   445	            // Clean remaining temp files
   446	            for (const tmpPath of tmpFiles.values()) {
   447	                if (fs.existsSync(tmpPath)) try { fs.unlinkSync(tmpPath); } catch {}
   448	            }
   449	            return {
   450	                success: false,
   451	                editCount: edits.length,
   452	                fileCount: editsByFile.size,
   453	                files: [],
   454	                error: `ACID commit aborted: ${(fatalErr as Error).message}. ${renamedFiles.length} files reverted from backup.`,
   455	            };
   456	        }
   457	    } else {
   458	        for (const filePath of vfs.keys()) {
   459	            writtenFiles.push(path.relative(projectRoot, filePath));
   460	        }
   461	    }
   462	
   463	    return {
   464	        success: true,
   465	        editCount: edits.length,
   466	        fileCount: editsByFile.size,
   467	        files: writtenFiles,
   468	        oldRawCodes,
   469	        vfs,
   470	    };
   471	}

   475	/**
   476	 * Surgically edit a single symbol in a file by name.
   477	 *
   478	 * 1. Parse file to find the target AST node
   479	 * 2. Use exact byte indices from tree-sitter (no indexOf)
   480	 * 3. Auto-indent the replacement to match original context
   481	 * 4. Splice: before + newCode + after (supports replace/insert_before/insert_after)
   482	 * 5. Validate the result with tree-sitter
   483	 * 6. Write only if syntax is valid
   484	 */
   485	export async function semanticEdit(
   486	    filePath: string,
   487	    symbolName: string,
   488	    newCode: string,
   489	    parser: ASTParser,
   490	    sandbox: AstSandbox,
   491	    mode: EditMode = "replace",
   492	    dryRun: boolean = false,
   493	): Promise<EditResult> {
   494	    // Read file
   495	    let content: string;
   496	    try {
   497	        content = readSource(filePath);
   498	    } catch (err) {
   499	        return {
   500	            success: false,
   501	            filePath,
   502	            symbolName,
   503	            oldLines: 0,
   504	            newLines: 0,
   505	            tokensAvoided: 0,
   506	            syntaxValid: false,
   507	            error: `Cannot read file: ${(err as Error).message}`,
   508	        };
   509	    }
   510	
   511	    // Parse to get AST chunks
   512	    const parseResult = await parser.parse(filePath, content);
   513	
   514	    if (parseResult.chunks.length === 0) {
   515	        return {
   516	            success: false,
   517	            filePath,
   518	            symbolName,
   519	            oldLines: 0,
   520	            newLines: 0,
   521	            tokensAvoided: 0,
   522	            syntaxValid: false,
   523	            error: "No symbols found in file. File may be empty or unsupported.",
   524	        };
   525	    }
   526	
   527	    // Find matching chunks by name (prefer AST symbolName, fallback to regex)
   528	    const matches: Array<{ chunk: ParsedChunk; name: string }> = [];
   529	    const allNames: string[] = [];
   530	
   531	    for (const chunk of parseResult.chunks) {
   532	        const name = chunk.symbolName || extractName(chunk);
   533	        if (name) allNames.push(`${name} (L${chunk.startLine}-L${chunk.endLine})`);
   534	        if (name === symbolName) {
   535	            matches.push({ chunk, name });
   536	        }
   537	    }
   538	
   539	    // Symbol not found
   540	    if (matches.length === 0) {
   541	        const lowerSymbol = symbolName.toLowerCase();
   542	        const fuzzy = parseResult.chunks
   543	            .map((c) => ({ chunk: c, name: c.symbolName || extractName(c) }))
   544	            .filter((c) => c.name.toLowerCase() === lowerSymbol);
   545	
   546	        if (fuzzy.length > 0) {
   547	            return {
   548	                success: false,
   549	                filePath,
   550	                symbolName,
   551	                oldLines: 0,
   552	                newLines: 0,
   553	                tokensAvoided: 0,
   554	                syntaxValid: false,
   555	                error:
   556	                    `Symbol "${symbolName}" not found. Did you mean "${fuzzy[0].name}"? ` +
   557	                    `Available symbols: ${allNames.join(", ")}`,
   558	            };
   559	        }
   560	
   561	        // BONUS: JIT Heuristic - guess the intended symbol by comparing tokens
   562	        const codeTokens = new Set(newCode.match(/[a-zA-Z_]\w*/g) || []);
   563	        let bestChunk: ParsedChunk | null = null;
   564	        let bestScore = 0;

   568	            let score = 0;
   569	            for (const term of codeTokens) {
   570	                if (chunkTokens.has(term)) score++;
   571	            }
   572	            if (score > bestScore) {
   573	                bestScore = score;
   574	                bestChunk = chunk;
   575	            }
   576	        }

   581	                success: false,
   582	                filePath,
   583	                symbolName,
   584	                oldLines: 0,
   585	                newLines: 0,
   586	                tokensAvoided: 0,
   587	                syntaxValid: false,
   588	                error:

   590	                    `Please retry using: symbol:"${guessedName}"`,
   591	            };
   592	        }
   593	
   594	        return {
   595	            success: false,
   596	            filePath,
   597	            symbolName,
   598	            oldLines: 0,
   599	            newLines: 0,
   600	            tokensAvoided: 0,
   601	            syntaxValid: false,

   603	        };

   607	    if (matches.length > 1) {
   608	        const locs = matches
   609	            .map((m) => `L${m.chunk.startLine}-L${m.chunk.endLine}`)
   610	            .join(", ");
   611	        return {
   612	            success: false,
   613	            filePath,
   614	            symbolName,
   615	            oldLines: 0,
   616	            newLines: 0,
   617	            tokensAvoided: 0,
   618	            syntaxValid: false,

   620	                `Multiple symbols named "${symbolName}" found at: ${locs}. ` +
   621	                `Use nreki_navigate action:"definition" to identify the correct one.`,
   622	        };

   626	
   627	    const { chunk } = matches[0];
   628	    const rawCode = chunk.rawCode;

   630

   632	    let newContent: string;
   633	    try {
   634	        newContent = applySemanticSplice(
   635	            content,
   636	            {
   637	                startIndex: chunk.startIndex,
   638	                endIndex: chunk.endIndex,
   639	                rawCode: chunk.rawCode,
   640	                symbolName: chunk.symbolName || extractName(chunk),
   641	                startLine: chunk.startLine,
   642	            },
   643	            newCode,
   644	            mode,
   645	        );
   646	    } catch (err) {
   647	        return {
   648	            success: false,
   649	            filePath,
   650	            symbolName,
   651	            oldLines: rawCode.split("\n").length,
   652	            newLines: newCode.split("\n").length,
   653	            tokensAvoided: 0,
   654	            syntaxValid: false,
   655	            error: (err as Error).message,
   656	        };
   657	    }
   658	
   659	    // Validate syntax of the edited file
   660	    const language = detectLanguage(filePath);
   661	    if (!language) {
   662	        return {
   663	            success: false,
   664	            filePath,
   665	            symbolName,
   666	            oldLines: rawCode.split("\n").length,
   667	            newLines: newCode.split("\n").length,
   668	            tokensAvoided: 0,
   669	            syntaxValid: false,
   670	            error: "Unsupported file type for syntax validation.",
   671	        };
   672	    }
   673	
   674	    await sandbox.initialize();

   676	
   677	    if (!validation.valid) {
   678	        const errDetails = validation.errors
   679	            .slice(0, 5)
   680	            .map(
   681	                (e) =>
   682	                    `  Line ${e.line}, Col ${e.column}: ${e.context.split("\n")[0].trim()}`,
   683	            )
   684	            .join("\n");
   685	
   686	        return {
   687	            success: false,
   688	            filePath,
   689	            symbolName,
   690	            oldLines: rawCode.split("\n").length,
   691	            newLines: newCode.split("\n").length,
   692	            tokensAvoided: 0,
   693	            syntaxValid: false,
   694	            error:
   695	                `Syntax error in edited code - file NOT modified:\n${errDetails}\n\n${validation.suggestion}`,
   696	        };
   697	    }
   698	
   699	    // Save backup and write to disk ONLY if not dry run
   700	    if (!dryRun) {
   701	        try {
   702	            saveBackup(process.cwd(), filePath);
   703	        } catch {
   704	            // Non-fatal: don't block the edit if backup fails
   705	        }
   706	        fs.writeFileSync(filePath, newContent, "utf-8");
   707	    }
   708	
   709	    // Tokens avoided: without NREKI Claude reads full file + sends old symbol code.
   710	    // With NREKI: only sends newCode. Savings = (fullFile + oldSymbol) - newCode.
   711	    const fullFileTokens = Embedder.estimateTokens(content);
   712	    const symbolTokens = Embedder.estimateTokens(rawCode);
   713	    const newCodeTokens = Embedder.estimateTokens(newCode);
   714	    const tokensAvoided = Math.max(0, fullFileTokens + symbolTokens - newCodeTokens);
   715	
   716	    return {
   717	        success: true,
   718	        filePath,
   719	        symbolName,
   720	        oldLines,
   721	        newLines: newCode.split("\n").length,
   722	        tokensAvoided,
   723	        syntaxValid: true,
   724	        oldRawCode: rawCode,
   725	        newContent,
   726	    };
   727	}

    98	/**
    99	 * Context Heartbeat - Session State Re-injection.
   100	 *
   101	 * Re-injects a 4-layer session state every ~15 tool calls to survive
   102	 * Claude Code's context compaction. Injects memory ABOVE the tool result
   103	 * (respects U-shaped attention curve in transformer models).
   104	 *
   105	 * Only fires during read-only actions (read, search, map, status,
   106	 * definition, references, outline). Never during edit, undo, or filter_output.
   107	 *
   108	 * Layers:
   109	 * 1. Plan File - Anchored plan document (PLAN.md, schemas, constraints)
   110	 * 2. Scratchpad - Claude's progress notes
   111	 * 3. Recent Edits - Files modified in this session
   112	 * 4. Circuit Breaker - Active escalation alerts
   113	 */
   114	export function applyContextHeartbeat(
   115	    action: string,
   116	    response: McpToolResponse,
   117	    deps: RouterDependencies,
   118	): McpToolResponse {
   119	    if (response.isError || !response.content || response.content[0]?.type !== "text") {
   120	        return response;
   121	    }
   122	
   123	    // C4: Don't inject heartbeat during active circuit breaker escalation
   124	    if (deps.circuitBreaker.getState().escalationLevel >= 2) {
   125	        return response;
   126	    }
   127	
   128	    try {
   129	        const currentCalls = deps.circuitBreaker.getStats().totalToolCalls;
   130	        let lastInjectCalls = parseInt(
   131	            deps.engine.getMetadata("nreki_plan_last_inject") || "0",
   132	            10,
   133	        );
   134	
   135	        // FIX: Session restart detection - if counter reset, reset the injection marker
   136	        if (currentCalls <= lastInjectCalls) {
   137	            lastInjectCalls = 0;
   138	            deps.engine.setMetadata("nreki_plan_last_inject", "0");
   139	        }
   140	
   141	        if (currentCalls - lastInjectCalls >= 15) {
   142	            const safeActions = [
   143	                "read", "search", "map", "status",
   144	                "definition", "references", "outline",
   145	            ];
   146	
   147	            if (safeActions.includes(action)) {
   148	                let memoryPayload = "";
   149	
   150	                // LAYER 1: Plan File
   151	                const planPath = deps.engine.getMetadata("nreki_master_plan");
   152	                if (planPath && fs.existsSync(planPath)) {
   153	                    let planContent: string;
   154	                    try { planContent = readSource(planPath); } catch { planContent = ""; }
   155	                    if (planContent && planContent.length < 15000) {
   156	                        memoryPayload +=
   157	                            `=== PLAN FILE (${path.basename(planPath)}) ===\n` +
   158	                            `${planContent}\n\n`;
   159	                    } else {
   160	                        memoryPayload +=
   161	                            `=== PLAN FILE ===\n` +
   162	                            `[WARNING: Your plan file "${path.basename(planPath)}" exceeds 15,000 characters (${planContent.length.toLocaleString()} chars). ` +

   165	                            `nreki_guard action:"set_plan" text:"<shorter_plan_file>"]\n\n`;

   167	                }
   168	
   169	                // LAYER 2: Scratchpad
   170	                const scratchpad = deps.engine.getMetadata("nreki_scratchpad");
   171	                if (scratchpad) {
   172	                    memoryPayload +=
   173	                        `=== SCRATCHPAD (Your Notes) ===\n` +
   174	                        `${scratchpad}\n\n`;
   175	                }
   176	
   177	                // LAYER 2b: Pinned Rules
   178	                try {
   179	                    const pinnedText = getPinnedText(process.cwd());
   180	                    if (pinnedText) {
   181	                        memoryPayload += `${pinnedText}\n\n`;
   182	                    }
   183	                } catch {
   184	                    // getPinnedText may fail - skip gracefully
   185	                }
   186	
   187	                // LAYER 3: Recent Edits
   188	                const history = deps.circuitBreaker.getState().history;
   189	                const recentEdits = new Set<string>();
   190	                const scanWindow = Math.max(0, history.length - 15);
   191	                for (let i = history.length - 1; i >= scanWindow; i--) {
   192	                    const record = history[i];
   193	                    if (
   194	                        record.filePath &&
   195	                        !record.errorHash
   196	                    ) {
   197	                        recentEdits.add(path.basename(record.filePath));
   198	                    }
   199	                }
   200	                if (recentEdits.size > 0) {
   201	                    memoryPayload +=
   202	                        `=== RECENT EDITS ===\n` +
   203	                        `You recently modified: ${Array.from(recentEdits).join(", ")}.\n\n`;
   204	                }
   205

   207	                const cbState = deps.circuitBreaker.getState();
   208	                if (cbState.escalationLevel > 0) {
   209	                    const target =
   210	                        cbState.lastTrippedSymbol ||
   211	                        cbState.lastTrippedFile ||
   212	                        "a critical component";
   213	                    memoryPayload +=
   214	                        `=== CIRCUIT BREAKER ALERT (LEVEL ${cbState.escalationLevel}) ===\n` +
   215	                        `You are executing a "Break & Build" strategy on \`${target}\`.\n` +
   216	                        `Do not deviate until this is resolved.\n\n`;
   217	                }

   220	                if (memoryPayload.trim().length > 0) {
   221	                    const header =
   222	                        `=================================================================\n` +
   223	                        ` [NREKI CONTEXT HEARTBEAT]\n` +
   224	                        ` Context compaction detected. Restoring session state:\n` +

   264	// ─── Rate Limiter (Token Bucket) ────────────────────────────────────
   265	//
   266	// Prevents hyperactive agents from overwhelming the kernel with
   267	// rapid-fire edits. Only applied to mutating operations (edit, batch_edit).
   268	// Reads and navigation are unrestricted.
   269	
   270	class TokenBucket {
   271	    private tokens: number;
   272	    private lastRefill: number;
   273	    constructor(
   274	        private readonly capacity: number,
   275	        private readonly refillRate: number, // tokens per second
   276	    ) {
   277	        this.tokens = capacity;
   278	        this.lastRefill = Date.now();
   279	    }
   280	
   281	    tryConsume(): boolean {
   282	        const now = Date.now();
   283	        const elapsed = (now - this.lastRefill) / 1000;
   284	        this.tokens = Math.min(this.capacity, this.tokens + elapsed * this.refillRate);
   285	        this.lastRefill = now;
   286	        if (this.tokens >= 1) {
   287	            this.tokens -= 1;
   288	            return true;
   289	        }
   290	        return false;
   291	    }
   292	}
