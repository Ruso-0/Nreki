   410	    /**
   411	     * Two-Phase Atomic Commit with physical rollback (P2, P18, P27, P32).
   412	     * Phase 1: Backup existing files to .bak
   413	     * Phase 2: Write new content via temp+rename
   414	     * Phase 3: Clean backups on success
   415	     * On failure: restore all .bak files
   416	     */
   417	    public async commitToDisk(): Promise<void> {
   418	        return this.mutex.withLock(async () => {
   419	        const physicalUndoLog: { target: string; backup: string | null }[] = [];
   420	        const createdTmps: string[] = [];
   421	
   422	        // Scope writes to files mutated in the current transaction only.
   423	        // Prevents stale VFS entries from prior intercepts leaking into this commit.
   424	        const filesToCommit = new Set(this.mutatedFiles);
   425	
   426	        try {
   427	            // PHASE 1: Physical backup
   428	            for (const posixPath of filesToCommit) {
   429	                const osPath = path.normalize(posixPath);
   430	                if (fs.existsSync(osPath)) {
   431	                    const txDir = path.join(this.projectRoot, ".nreki", "transactions");
   432	                    if (!fs.existsSync(txDir)) fs.mkdirSync(txDir, { recursive: true });
   433	                    const bak = path.join(txDir, `${crypto.randomBytes(4).toString("hex")}.bak`);
   434	                    fs.copyFileSync(osPath, bak);
   435	                    physicalUndoLog.push({ target: osPath, backup: bak });
   436	                } else {
   437	                    physicalUndoLog.push({ target: osPath, backup: null });
   438	                }
   439	            }
   440	
   441	            // WAL: Write intent log before destructive Phase 2.
   442	            // If the process crashes between Phase 1 (backup) and Phase 2 (write),
   443	            // boot() will find this WAL and automatically restore from backups.
   444	            const txDir = path.join(this.projectRoot, ".nreki", "transactions");
   445	            const walPath = path.join(txDir, "wal.json");
   446	            try {
   447	                if (!fs.existsSync(txDir)) fs.mkdirSync(txDir, { recursive: true });
   448	                // Atomic write via temp+rename. writeFileSync overwrites in-place
   449	                // and is NOT atomic — OOM/crash mid-write truncates the WAL.
   450	                // rename() is atomic on POSIX (single inode pointer swap).
   451	                const walTmpPath = `${walPath}.tmp`;
   452	                fs.writeFileSync(walTmpPath, JSON.stringify({
   453	                    status: "pending",
   454	                    ts: new Date().toISOString(),
   455	                    files: physicalUndoLog.map(l => ({ target: l.target, backup: l.backup })),
   456	                }), "utf-8");
   457	                fs.renameSync(walTmpPath, walPath);
   458	            } catch (walErr) {
   459	                // WAL write failure is non-fatal but must be logged — proceeding
   460	                // without crash safety means a power failure during Phase 2 is unrecoverable.
   461	                logger.warn(`WAL write failed: ${walErr}. Proceeding without crash recovery safety.`);
   462	            }
   463	
   464	            // PHASE 2: Destructive writes
   465	            for (const posixPath of filesToCommit) {
   466	                const content = this.vfs.get(posixPath) ?? null;
   467	                const osPath = path.normalize(posixPath);
   468	                if (content === null) {
   469	                    if (fs.existsSync(osPath)) fs.unlinkSync(osPath);
   470	                } else {
   471	                    const dir = path.dirname(osPath);
   472	                    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
   473	                    const tmp = `${osPath}.nreki-${crypto.randomBytes(4).toString("hex")}.tmp`;
   474	                    fs.writeFileSync(tmp, content, "utf-8");
   475	                    createdTmps.push(tmp);
   476	                    fs.renameSync(tmp, osPath);
   477	                    createdTmps.pop(); // Rename succeeded — no longer orphan
   478	                }
   479	            }
   480	
   481	            // P18: VFS clear and rebuild
   482	            // C-02: Save VFS state before clearing so we can restore on rebuild failure
   483	            const savedVfs = new Map(this.vfs);
   484	            const savedClock = new Map(this.vfsClock);
   485	            const savedDirs = new Set(this.vfsDirectories);
   486	            this.vfs.clear();
   487	            this.vfsClock.clear();
   488	            this.vfsDirectories.clear();
   489	            this.tsBackend.purgeCache();
   490	            try {
   491	                this.tsBackend.updateProgram();
   492	                // DR/LS synced via tsBackend
   493	                this.tsBackend.captureBaseline(undefined, this.mode);
   494	            } catch (rebuildErr) {
   495	                // A-10: Force full rebuild on next operation
   496	                this.tsBackend.purgeCache();
   497	                // Restore VFS to pre-clear state for consistency with rolled-back disk
   498	                for (const [k, v] of savedVfs) this.vfs.set(k, v);
   499	                for (const [k, v] of savedClock) this.vfsClock.set(k, v);
   500	                for (const d of savedDirs) this.vfsDirectories.add(d);
   501	                throw rebuildErr;
   502	            }
   503	
   504	            // Persist build state for next session (warm boot)
   505	            try {
   506	                this.tsBackend.tsBuilder!.emit(
   507	                    undefined,
   508	                    (fileName, text) => {
   509	                        // Only write .tsbuildinfo, not .js/.d.ts
   510	                        if (fileName.endsWith(".tsbuildinfo")) {
   511	                            const dir = path.dirname(fileName);
   512	                            if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
   513	                            fs.writeFileSync(fileName, text, "utf-8");
   514	                        }
   515	                    },
   516	                    undefined,
   517	                    true, // emitOnlyDtsFiles - suppresses .js output
   518	                );
   519	                this.writeTsVersionGuard();
   520	            } catch { /* Non-fatal: next boot will cold-start */ }
   521	
   522	            // A-01: Phase 3 AFTER successful rebuild - backups survive if rebuild throws
   523	            for (const log of physicalUndoLog) {
   524	                if (log.backup && fs.existsSync(log.backup)) fs.unlinkSync(log.backup);
   525	            }
   526	            // WAL: Transaction complete — delete intent log
   527	            try { if (fs.existsSync(walPath)) fs.unlinkSync(walPath); } catch { /* best effort */ }
   528	
   529	            // Release stale AST versions from DocumentRegistry
   530	            this.tsBackend.releaseMutatedDocuments(this.mutatedFiles);
   531	            this.mutatedFiles.clear();
   532	
   533	            logger.info("Atomic commit materialized. Disk synchronized.");
   534	        } catch (error) {
   535	            // PHYSICAL ROLLBACK
   536	            logger.error(`OS write failure! Physical rollback: ${error}`);
   537	
   538	            // AUDIT FIX: Clean orphan .tmp files written before the crash
   539	            for (const tmpPath of createdTmps) {
   540	                try { if (fs.existsSync(tmpPath)) fs.unlinkSync(tmpPath); } catch { /* best effort */ }
   541	            }
   542	
   543	            for (const log of physicalUndoLog) {
   544	                try {
   545	                    if (log.backup) {
   546	                        if (fs.existsSync(log.backup)) fs.renameSync(log.backup, log.target);
   547	                    } else {
   548	                        if (fs.existsSync(log.target)) fs.unlinkSync(log.target);
   549	                    }
   550	                } catch { /* Cascade panic - best effort */ }
   551	            }
   552	
   553	            // VFS ZOMBIE FIX: Purge RAM state that never reached disk.
   554	            // Without this, the next readFile() serves ghost content
   555	            // desynchronized from the real filesystem.
   556	            this.vfs.clear();
   557	            this.vfsClock.clear();
   558	            this.vfsDirectories.clear();
   559	            this.logicalTime += 1000;
   560	
   561	            // Force cold rebuild from disk reality
   562	            this.tsBackend.purgeCache();
   563	            try {
   564	                this.tsBackend.updateProgram();
   565	                // DR/LS synced via tsBackend
   566	                this.tsBackend.captureBaseline(undefined, this.mode);
   567	            } catch {
   568	                // If rebuild also fails, mark kernel for full reconstruction
   569	                this.tsBackend.purgeCache(true);
   570	            }
   571	
   572	            throw new Error(`[NREKI] Physical ACID commit failed. Repository and VFS restored. Reason: ${error}`);
   573	        }
   574	        });
   575	    }

   577	    /** Emergency rollback - purge all staged changes (P3). */
   578	    public async rollbackAll(): Promise<void> {
   579	        return this.mutex.withLock(async () => {
   580	            this.vfs.clear();
   581	            this.vfsClock.clear();
   582	            this.vfsDirectories.clear();
   583	            this.initConfig(); // Re-sync rootNames from disk
   584	            // BUG 1: In hologram mode, re-filter rootNames
   585	            if (this.mode === "hologram") {
   586	                if (this.jitMode) {
   587	                    // SURGICAL JIT CACHE INVALIDATION:
   588	                    // jitClassifyFile() reads from DISK, not VFS. Classifications
   589	                    // remain valid after rollback because disk content is unchanged.
   590	                    // Only invalidate files that were being edited in this transaction.
   591	                    // This preserves the ~1.94s of accumulated JIT work.
   592	                    for (const target of this.currentEditTargets) {
   593	                        this.jitClassifiedCache.delete(target);
   594	                        this.prunedFiles.delete(target);
   595	                        this.shadowContent.delete(target);
   596	                    }
   597	                    this.buildShadowLookups();
   598	                    this.currentEditTargets.clear();
   599	                    this.rootNames = new Set(
   600	                        [...this.rootNames].filter(f => /\.d\.[mc]?ts$/i.test(f)),
   601	                    );
   602	                    this.tsBackend.rootNames = this.rootNames;
   603	                } else if (this.ambientFiles.length > 0) {
   604	                    // Eager mode: re-filter to ambient-only
   605	                    const ambientSet = new Set(
   606	                        this.ambientFiles.map(f => this.toPosix(path.resolve(this.projectRoot, f))),
   607	                    );
   608	                    this.rootNames = new Set(
   609	                        [...this.rootNames].filter(f => ambientSet.has(f)),
   610	                    );
   611	                    this.tsBackend.rootNames = this.rootNames;
   612	                    this.buildShadowLookups();
   613	                }
   614	            }
   615	            // WARM-PATH: Advance clock to invalidate cached files.
   616	            // Only destroy builderProgram if it doesn't exist yet.
   617	            this.logicalTime += 1000;
   618	            this.tsBackend.updateProgram();
   619	                // DR/LS synced via tsBackend
   620	            // Release stale AST versions from DocumentRegistry
   621	            this.tsBackend.releaseMutatedDocuments(this.mutatedFiles);
   622	            logger.warn("Hard rollback executed. VFS purged.");
   623	        });
   624	    }

   646	    /**
   647	     * Show which files depend on a symbol before the agent edits it.
   648	     * Uses LanguageService.findReferences(). Cost: ~20ms per call.
   649	     */
   650	    public predictBlastRadius(targetFile: string, symbolName: string): { safe: boolean; report: string } {
   651	        if (this.mode === "hologram") {
   652	            throw new Error(
   653	                "[NREKI] Cannot use predictBlastRadius in hologram mode. " +
   654	                "Use Layer 1 AST navigator for reference queries.",
   655	            );
   656	        }
   657	        if (!this.booted || !this.tsBackend.tsLanguageService || !this.tsBackend.tsBuilder) {
   658	            throw new Error("[NREKI] Kernel or Language Service not booted.");
   659	        }
   660	
   661	        const posixPath = this.toPosix(path.resolve(this.projectRoot, targetFile));
   662	        const program = this.tsBackend.tsLanguageService.getProgram();
   663	        if (!program) throw new Error("[NREKI] Could not get program from LanguageService");
   664	        const sourceFile = program.getSourceFile(posixPath);
   665	        if (!sourceFile) throw new Error(`[NREKI] File not tracked: ${targetFile}`);
   666	
   667	        // 1. Find the declaration's byte offset
   668	        let targetPos = -1;
   669	        const findNode = (node: ts.Node) => {
   670	            if (ts.isIdentifier(node) && node.text === symbolName) {
   671	                const parent = node.parent;
   672	                if (parent && (
   673	                    ts.isFunctionDeclaration(parent) || ts.isClassDeclaration(parent) ||
   674	                    ts.isMethodDeclaration(parent) || ts.isVariableDeclaration(parent) ||
   675	                    ts.isInterfaceDeclaration(parent) || ts.isTypeAliasDeclaration(parent) ||
   676	                    ts.isPropertySignature(parent) || ts.isMethodSignature(parent)
   677	                ) && (parent as ts.NamedDeclaration).name === node) {
   678	                    targetPos = node.getStart(sourceFile);
   679	                    return;
   680	                }
   681	            }
   682	            if (targetPos === -1) ts.forEachChild(node, findNode);
   683	        };
   684	        findNode(sourceFile);
   685	
   686	        if (targetPos === -1) {
   687	            throw new Error(`[NREKI] Declaration for '${symbolName}' not found in ${targetFile}`);
   688	        }
   689	
   690	        // 2. Fire the LanguageService reference scanner
   691	        const references = this.tsBackend.tsLanguageService.findReferences(posixPath, targetPos);
   692	        if (!references || references.length === 0) {
   693	            return {
   694	                safe: true,
   695	                report: `No external dependents found for ${symbolName}. Safe to change.`,
   696	            };
   697	        }
   698	
   699	        const dependents = new Map<string, Array<{ line: number; context: string; reason: string }>>();
   700	        let totalUsages = 0;
   701	
   702	        for (const ref of references) {
   703	            for (const curr of ref.references) {
   704	                if (curr.isDefinition) continue;
   705	
   706	                const refFile = this.toPosix(path.resolve(this.projectRoot, curr.fileName));
   707	                if (refFile === posixPath) continue;
   708	
   709	                const refSourceFile = program.getSourceFile(refFile);
   710	                if (!refSourceFile) continue;
   711	
   712	                const { line } = ts.getLineAndCharacterOfPosition(refSourceFile, curr.textSpan.start);
   713	                const lineText = (refSourceFile.text.split("\n")[line] ?? "").trim();
   714	
   715	                // Extract WHY this usage matters (Demand Inference)
   716	                let reason = "Structural usage";
   717	                let usageNode: ts.Node | undefined;
   718	                const findUsage = (n: ts.Node) => {
   719	                    if (curr.textSpan.start >= n.getStart(refSourceFile) && curr.textSpan.start < n.getEnd()) {
   720	                        usageNode = n;
   721	                        ts.forEachChild(n, findUsage);
   722	                    }
   723	                };
   724	                findUsage(refSourceFile);
   725	
   726	                if (usageNode && usageNode.parent) {
   727	                    const p = usageNode.parent;
   728	                    if (ts.isCallExpression(p)) reason = "Function call (expects specific arguments/return type)";
   729	                    else if (ts.isPropertyAccessExpression(p)) reason = "Property access";
   730	                    else if (ts.isBinaryExpression(p)) reason = "Binary operation / Assignment";
   731	                    else if (ts.isVariableDeclaration(p)) reason = "Assigned to explicitly typed variable";
   732	                    else if (ts.isTypeReferenceNode(p)) reason = "Type contract (Type Reference)";
   733	                    else if (ts.isImportSpecifier(p) || ts.isImportClause(p)) reason = "Module import";
   734	                    else if (ts.isExportSpecifier(p) || ts.isExportAssignment(p)) reason = "Module export";
   735	                    else if (ts.isJsxOpeningElement(p) || ts.isJsxSelfClosingElement(p)) reason = "JSX Component instantiation";
   736	                }
   737	
   738	                const relPath = path.relative(this.projectRoot, curr.fileName).replace(/\\/g, "/");
   739	                const arr = dependents.get(relPath) || [];
   740	                arr.push({ line: line + 1, context: lineText, reason });
   741	                dependents.set(relPath, arr);
   742	                totalUsages++;
   743	            }
   744	        }
   745	
   746	        if (dependents.size === 0) {
   747	            return {
   748	                safe: true,
   749	                report: `${symbolName} is only used internally in ${path.relative(this.projectRoot, posixPath)}. Safe to change.`,
   750	            };
   751	        }
   752	
   753	        // 3. Format tactical report for the LLM
   754	        const lines: string[] = [
   755	            `${totalUsages} references across ${dependents.size} files will break if you change this signature.`,
   756	            `Dependent files:\n`,
   757	        ];
   758	
   759	        for (const [file, usages] of dependents.entries()) {
   760	            lines.push(`${file} (${usages.length} usages)`);
   761	            for (const u of usages) {
   762	                lines.push(`  L${u.line}: ${u.context} (${u.reason})`);
   763	            }
   764	        }
   765	
   766	        lines.push(
   767	            `\nUse nreki_code action:"batch_edit" to update all dependent files in the same transaction.`,
   768	        );
   769	
   770	        return { safe: false, report: lines.join("\n") };
   771	    }

   855	    /**
   856	     * JIT classify a single .ts/.tsx file and populate shadow data structures.
   857	     * Returns true if file was prunable (shadow generated), false otherwise.
   858	     * Synchronous: reads file from disk, classifies with pre-loaded tree-sitter parser.
   859	     */
   860	    private jitClassifyFile(tsPath: string): boolean {
   861	        if (this.jitClassifiedCache.has(tsPath)) return this.prunedTsLookup.has(tsPath);
   862	        if (!this.jitParser || !this.jitTsLanguage || !this.jitClassifyFn) return false;
   863	
   864	        this.jitClassifiedCache.add(tsPath);
   865	
   866	        // SIZE GUARD: Skip large auto-generated files (GraphQL codegen,
   867	        // Prisma client, protobuf output) that would block the event loop.
   868	        try {
   869	            const stat = fs.statSync(path.normalize(tsPath));
   870	            if (stat.size > NrekiKernel.JIT_MAX_FILE_SIZE) return false;
   871	        } catch { return false; }
   872	
   873	        let content: string;
   874	        try { content = readSource(path.normalize(tsPath)); }
   875	        catch { return false; }
   876	
   877	        const result = this.jitClassifyFn(tsPath, content, this.jitParser, this.jitTsLanguage);
   878	        if (result.prunable && result.shadow) {
   879	            this.prunedFiles.add(tsPath);
   880	            this.shadowContent.set(tsPath, result.shadow);
   881	            const dtsPath = tsPath.replace(/\.([mc]?)tsx?$/, ".d.$1ts");
   882	            this.shadowDtsLookup.add(dtsPath);
   883	            this.prunedTsLookup.add(tsPath);
   884	            return true;
   885	        }
   886	        return false;
   887	    }

   894	            this.shadowDtsLookup.add(dtsPath);

   896	        }

   899	    // ─── LSP Sidecar Management ─────────────────────────────────────
   900	
   901	    /**
   902	     * Compensatory Rollback — heal sidecar VFS on ACID abort.
   903	     *
   904	     * When `interceptAtomicBatch` rejects a transaction, the TypeScript VFS
   905	     * is rolled back via the `rollbackState` map. But LSP sidecars (gopls,
   906	     * pyright) have their OWN VFS in RAM. If we don't re-inject the original
   907	     * content, the sidecar's next `validateEdits` call will run against the
   908	     * phantom (rejected) code, causing a brain-split.
   909	     *
   910	     * This method sends `didChange` with the original content for every file
   911	     * that was modified in the aborted transaction. Fire-and-forget — we don't
   912	     * await the settle because the rollback is best-effort and must not block
   913	     * the error response to the LLM.
   914	     */
   915	    private async rollbackSidecars(
   916	        sidecarEdits: Map<LspSidecarBase, Array<{filePath: string; content: string | null}>>,
   917	        rollbackState: Map<string, {content: string | null | undefined; time: Date | undefined; wasInRoot: boolean}>,
   918	    ): Promise<void> {
   919	        if (sidecarEdits.size === 0) return;
   920	
   921	        const promises: Promise<void | NrekiStructuredError[]>[] = [];
   922	        for (const [sidecar, scEdits] of sidecarEdits) {
   923	            if (sidecar.isDead) continue;
   924	            const rollbacks: Array<{filePath: string; content: string | null}> = [];
   925	
   926	            for (const edit of scEdits) {
   927	                const posixPath = this.toPosix(path.resolve(this.projectRoot, edit.filePath));
   928	                const state = rollbackState.get(posixPath);
   929	
   930	                let originalContent: string | null = null;
   931	                if (state && state.content !== undefined) {
   932	                    originalContent = state.content;
   933	                } else {
   934	                    // File didn't exist in VFS, try reading from disk
   935	                    try { originalContent = fs.readFileSync(edit.filePath, "utf-8"); } catch { originalContent = null; }
   936	                }
   937	
   938	                rollbacks.push({ filePath: edit.filePath, content: originalContent });
   939	            }
   940	
   941	            if (rollbacks.length > 0) {
   942	                promises.push(
   943	                    sidecar.validateEdits(rollbacks).catch(() => {
   944	                        sidecar.isDead = true;
   945	                    })
   946	                );
   947	            }
   948	        }
   949	        await Promise.all(promises);
   950	    }
