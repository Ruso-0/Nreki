    50	/**
    51	 * Resolve and validate a file path, ensuring it stays within the workspace root.
    52	 *
    53	 * Security layers:
    54	 *   1. Path traversal check (../ sequences)
    55	 *   2. Symlink resolution (prevents symlink escapes)
    56	 *   3. Sensitive file blocklist (.env, .ssh, .pem, etc.)
    57	 *
    58	 * @param workspaceRoot - The allowed root directory
    59	 * @param inputPath - User-supplied file path (relative or absolute)
    60	 * @returns The resolved absolute path within the workspace
    61	 * @throws Error if the path resolves outside the workspace root or is sensitive
    62	 */
    63	export function safePath(workspaceRoot: string, inputPath: string): string {
    64	    // Normalize backslashes to forward slashes so traversal detection works on Linux/macOS
    65	    // NFC normalization: macOS uses NFD (decomposed), Linux uses NFC.
    66	    // Without this, an attacker can bypass .env blocklist with NFD-encoded paths.
    67	    const normalized = inputPath.normalize("NFC").replace(/\\/g, "/");
    68	    const resolved = path.resolve(workspaceRoot, normalized);
    69	    const resolvedRoot = path.resolve(workspaceRoot);
    70	
    71	    if (!resolved.startsWith(resolvedRoot + path.sep) && resolved !== resolvedRoot) {
    72	        throw new Error(`Path traversal blocked: ${inputPath}`);
    73	    }
    74	
    75	    // H-02: Block operating on the workspace root itself
    76	    if (resolved === resolvedRoot) {
    77	        throw new Error(`Cannot operate on workspace root directly: ${inputPath}`);
    78	    }
    79	
    80	    // Resolve symlinks to detect symlink escapes.
    81	    // Both resolved and root must be real-pathed so that systems where the
    82	    // workspace sits under a symlinked directory (e.g. macOS /tmp → /private/tmp)
    83	    // don't produce false-positive "escape" errors.
    84	    try {
    85	        const realPath = fs.realpathSync(resolved);
    86	        let realRoot: string;
    87	        try {
    88	            realRoot = fs.realpathSync(resolvedRoot);
    89	        } catch {
    90	            realRoot = resolvedRoot;
    91	        }
    92	        if (!realPath.startsWith(realRoot + path.sep) && realPath !== realRoot) {
    93	            throw new Error(`Symlink escape blocked: ${inputPath} resolves outside workspace`);
    94	        }
    95	    } catch (err) {
    96	        if ((err as NodeJS.ErrnoException).code === "ENOENT") {
    97	            // File doesn't exist yet — resolve parent chain to catch symlink escapes
    98	            let parent = path.dirname(resolved);
    99	            // Walk up until we find an existing directory
   100	            while (!fs.existsSync(parent) && parent !== path.dirname(parent)) {
   101	                parent = path.dirname(parent);
   102	            }
   103	            try {
   104	                const realParent = fs.realpathSync(parent);
   105	                let realRoot: string;
   106	                try {
   107	                    realRoot = fs.realpathSync(resolvedRoot);
   108	                } catch {
   109	                    realRoot = resolvedRoot;
   110	                }
   111	                if (!realParent.startsWith(realRoot + path.sep) && realParent !== realRoot) {
   112	                    throw new Error(`Symlink escape blocked in parent directory: ${inputPath}`);
   113	                }
   114	            } catch (parentErr) {
   115	                if ((parentErr as Error).message.includes("Symlink escape blocked")) {
   116	                    throw parentErr;
   117	                }
   118	                // FAIL-CLOSED: Any I/O error during security validation MUST deny access.
   119	                // EACCES (no permission), ELOOP (symlink loop), or any other OS error
   120	                // means we cannot verify the path is safe. Block it.
   121	                throw new Error(`Path validation failed securely during parent resolution: ${(parentErr as Error).message}`);
   122	            }
   123	        } else {
   124	            throw err;
   125	        }
   126	    }
   127	
   128	    // Block access to sensitive files
   129	    if (isSensitivePath(resolved)) {
   130	        throw new Error(`Access to sensitive file blocked: ${inputPath}`);
   131	    }
   132	
   133	    return resolved;
   134	}

   157	    public static computeDelta(
   158	        pre: SpectralResult,
   159	        post: SpectralResult
   160	    ): SpectralDelta {
   161	
   162	        // Φ: Topological Entropy Index
   163	        // When N stays constant (ghost/expansion): Φ = λ₂ * density, density = 2V/(N*(N-1))
   164	        // When N decreases (decoupling): Φ = λ₂ / N_AST (original formula)
   165	        let phiPre: number, phiPost: number;
   166	
   167	        if (pre.nodeCount <= post.nodeCount) {
   168	            // Ghost / Expansion case: nodes stay or grow but lose edges.
   169	            // Density formula Φ = λ₂ * (2V / (N*(N-1))) detects dilution.
   170	            const denPre = pre.nodeCount > 1 ? pre.nodeCount * (pre.nodeCount - 1) : 1;
   171	            phiPre = pre.nodeCount > 0 ? (pre.fiedlerValue * 2 * pre.volume) / denPre : 0;
   172	            const denPost = post.nodeCount > 1 ? post.nodeCount * (post.nodeCount - 1) : 1;
   173	            phiPost = post.nodeCount > 0 ? (post.fiedlerValue * 2 * post.volume) / denPost : 0;
   174	        } else {
   175	            // Decoupling case: nodes were deleted from AST.
   176	            // Original formula respects N_AST reduction.
   177	            phiPre = pre.nodeCount > 0 ? pre.fiedlerValue / pre.nodeCount : 0;
   178	            phiPost = post.nodeCount > 0 ? post.fiedlerValue / post.nodeCount : 0;
   179	        }
   180	        const normalizedFiedlerDrop = phiPre - phiPost;
   181	        const dropRatio = phiPre > 0 ? normalizedFiedlerDrop / phiPre : 0;
   182	        const volumeDrop = pre.volume - post.volume;
   183	
   184	        // Adaptive epsilon with hard bounds [0.10, 0.30]
   185	        // N_AST=3 → ε≈0.30 (ghost in K3 = 33% drop, always caught)
   186	        // N_AST=50 → ε≈0.15 (standard)
   187	        // N_AST=500 → ε≈0.10 (floor, never less sensitive than 10%)
   188	        const baseEpsilon = 0.15;
   189	        const scaleFactor = Math.sqrt(50 / Math.max(1, pre.nodeCount));
   190	        const epsilonDynamic = Math.max(0.10, Math.min(0.30, baseEpsilon * scaleFactor));
   191	
   192	        let verdict: SpectralDelta["verdict"];
   193	
   194	        if (volumeDrop > 0 && dropRatio > epsilonDynamic) {
   195	            verdict = "REJECTED_ENTROPY";
   196	        } else if (volumeDrop <= 0 && normalizedFiedlerDrop > 0) {
   197	            verdict = "APPROVED_DECOUPLING";
   198	        } else {
   199	            verdict = "APPROVED";
   200	        }
   201	
   202	        return {
   203	            fiedlerPre: pre.fiedlerValue,
   204	            fiedlerPost: post.fiedlerValue,
   205	            volumePre: pre.volume,
   206	            volumePost: post.volume,
   207	            normalizedFiedlerDrop,
   208	            volumeDrop,
   209	            verdict,
   210	        };
   211	    }

   311	    public static analyzeTopology(N: number, edges: SparseEdge[]): {
   312	        fiedler: number; volume: number;
   313	        v2?: Float64Array; lambda3?: number; v3?: Float64Array;
   314	    } {
   315	        if (N <= 1) return { fiedler: 0, volume: 0 };
   316	
   317	        // Size guard: power iteration is O(N² × 150 iterations).
   318	        // For N>5000 this could block the event loop for hundreds of ms.
   319	        // Return a structural estimate instead.
   320	        if (N > 25000) {
   321	            let volume = 0;
   322	            for (const e of edges) volume += e.weight;
   323	            const avgDegree = (2 * edges.length) / N;
   324	            return { fiedler: avgDegree * 0.5, volume };
   325	        }
   326	
   327	        // --- Edge deduplication (IDÉNTICO AL ORIGINAL - NO TOCAR) ---
   328	        const edgeMap = new Map<number, Map<number, number>>();
   329	
   330	        for (let i = 0; i < edges.length; i++) {
   331	            const e = edges[i];
   332	            if (e.u === e.v) continue;
   333	
   334	            const min = Math.min(e.u, e.v);
   335	            const max = Math.max(e.u, e.v);
   336	
   337	            let row = edgeMap.get(min);
   338	            if (!row) {
   339	                row = new Map<number, number>();
   340	                edgeMap.set(min, row);
   341	            }
   342	
   343	            const currentW = row.get(max) || 0;
   344	            if (e.weight > currentW) row.set(max, e.weight);
   345	        }
   346	
   347	        // --- CSR construction pass 1 (IDÉNTICO AL ORIGINAL - NO TOCAR) ---
   348	        const degree = new Float64Array(N);
   349	        const neighborCount = new Int32Array(N);
   350	        let maxDegree = 0;
   351	        let volume = 0;
   352	        let seed = N * 2654435761;
   353	
   354	        for (const [u, row] of edgeMap.entries()) {
   355	            for (const [v, w] of row.entries()) {
   356	                neighborCount[u]++; neighborCount[v]++;
   357	                degree[u] += w; degree[v] += w;
   358	                volume += w;
   359	                if (degree[u] > maxDegree) maxDegree = degree[u];
   360	                if (degree[v] > maxDegree) maxDegree = degree[v];
   361	                seed = ((seed << 5) - seed + (w * 1000 | 0)) | 0;
   362	            }
   363	        }
   364	
   365	        // Prefix sum → rowPtr (IDÉNTICO AL ORIGINAL - NO TOCAR)
   366	        const rowPtr = new Int32Array(N + 1);
   367	        for (let i = 0; i < N; i++) rowPtr[i + 1] = rowPtr[i] + neighborCount[i];
   368	        const nnz = rowPtr[N];
   369	
   370	        // --- CSR construction pass 2 (IDÉNTICO AL ORIGINAL - NO TOCAR) ---
   371	        const colIdx = new Int32Array(nnz);
   372	        const csrValues = new Float64Array(nnz);
   373	        for (let i = 0; i < N; i++) neighborCount[i] = rowPtr[i];
   374	
   375	        for (const [u, row] of edgeMap.entries()) {
   376	            for (const [v, w] of row.entries()) {
   377	                colIdx[neighborCount[u]] = v; csrValues[neighborCount[u]++] = w;
   378	                colIdx[neighborCount[v]] = u; csrValues[neighborCount[v]++] = w;
   379	            }
   380	        }
   381	
   382	        // ─── NUEVO: Capturar seed post-mutación + Power Iteration con Deflación ───
   383	
   384	        const dataSeed = seed; // Semilla POST-mutación exacta del original
   385	        const c = maxDegree * 2.0 + 1.0;
   386	
   387	        const powerIteration = (
   388	            deflateVectors: Float64Array[],
   389	            seedModifier: number
   390	        ): { val: number; vec: Float64Array } => {
   391	            const vec = new Float64Array(N);
   392	            let currentSeed = (dataSeed + seedModifier) | 0;
   393	            for (let i = 0; i < N; i++) {
   394	                currentSeed = (currentSeed * 1103515245 + 12345) | 0;
   395	                vec[i] = ((currentSeed >>> 16) & 0x7fff) / 32768.0 - 0.5;
   396	            }
   397	
   398	            const v_next = new Float64Array(N);
   399	            let mu = 0;
   400	            let prev_mu = -1;
   401	            let prevDelta = Infinity;
   402	
   403	            for (let iter = 0; iter < 150; iter++) {
   404	                // 1. Deflate trivial eigenvector (constant)
   405	                let sum = 0;
   406	                for (let i = 0; i < N; i++) sum += vec[i];
   407	                const mean = sum / N;
   408	                for (let i = 0; i < N; i++) vec[i] -= mean;
   409	
   410	                // 2. Gram-Schmidt orthogonalization against previous vectors
   411	                for (const dv of deflateVectors) {
   412	                    let dot = 0;
   413	                    for (let i = 0; i < N; i++) dot += vec[i] * dv[i];
   414	                    for (let i = 0; i < N; i++) vec[i] -= dot * dv[i];
   415	                }
   416	
   417	                // 3. L2 normalization
   418	                let normSq = 0;
   419	                for (let i = 0; i < N; i++) normSq += vec[i] * vec[i];
   420	                if (normSq < 1e-18) break;
   421	                const rNorm = 1.0 / Math.sqrt(normSq);
   422	                for (let i = 0; i < N; i++) vec[i] *= rNorm;
   423	
   424	                // 4. Fused SpMV: (cI - L)v + Rayleigh Quotient
   425	                let norm = 0;
   426	                mu = 0;
   427	                for (let i = 0; i < N; i++) {
   428	                    let Lv_i = degree[i] * vec[i];
   429	                    const end = rowPtr[i + 1];
   430	                    for (let k = rowPtr[i]; k < end; k++) {
   431	                        Lv_i -= csrValues[k] * vec[colIdx[k]];
   432	                    }
   433	                    const val = c * vec[i] - Lv_i;
   434	                    v_next[i] = val;
   435	                    norm += val * val;
   436	                    mu += vec[i] * val;
   437	                }
   438	
   439	                norm = Math.sqrt(norm);
   440	                if (norm < 1e-9) break;
   441	                // FIX: Dual convergence criterion.
   442	                // Check BOTH eigenvalue (mu) AND eigenvector stability.
   443	                // The eigenvalue converges quadratically faster than the
   444	                // eigenvector. Checking only mu can exit while the vector
   445	                // is still rotating, producing noisy v2/v3 components.
   446	                let maxVecDiff = 0;
   447	                for (let j = 0; j < N; j++) {
   448	                    const diff = Math.abs(vec[j] - v_next[j] / norm);
   449	                    if (diff > maxVecDiff) maxVecDiff = diff;
   450	                }
   451	                for (let i = 0; i < N; i++) vec[i] = v_next[i] / norm;
   452	
   453	                const delta = Math.abs(mu - prev_mu);
   454	                if (delta < 1e-7 && maxVecDiff < 1e-6) break;
   455	                // Divergence guard: if eigenvalue delta is growing instead of shrinking, bail out
   456	                if (iter > 20 && delta > prevDelta * 2) break;
   457	                prevDelta = delta;
   458	                prev_mu = mu;
   459	            }
   460	
   461	            // GAUGE FIXING (Phase Canonicalization)
   462	            // Lv = λv → if v is an eigenvector, -v is too.
   463	            // Power iteration converges to v or -v arbitrarily due to floating-point noise.
   464	            // To prevent a temporal neural network from seeing fictitious phase jumps between commits,
   465	            // we force deterministic orientation: the largest-magnitude component is always positive.
   466	            let maxAbs = -1;
   467	            let signMultiplier = 1;
   468	            for (let i = 0; i < N; i++) {
   469	                const absVal = Math.abs(vec[i]);
   470	                if (absVal > maxAbs) {
   471	                    maxAbs = absVal;
   472	                    signMultiplier = vec[i] < 0 ? -1 : 1;
   473	                }
   474	            }
   475	            if (signMultiplier === -1) {
   476	                for (let i = 0; i < N; i++) vec[i] *= -1;
   477	            }
   478	
   479	            return { val: Math.max(0, c - mu), vec };
   480	        };
   481	
   482	        // Extract λ₂ (Fiedler) — seedModifier=0 preserves original behavior
   483	        const res2 = powerIteration([], 0);
   484	
   485	        // Extract λ₃ — seedModifier=99991 ensures different subspace, deflating v₂
   486	        const res3 = powerIteration([res2.vec], 99991);
   487	
   488	        return {
   489	            fiedler: res2.val,
   490	            volume,
   491	            v2: res2.vec,
   492	            lambda3: res3.val,
   493	            v3: res3.vec,
   494	        };
   495	    }
   496	}
