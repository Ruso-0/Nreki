    16	export class AsyncMutex {
    17	    private queue: (() => void)[] = [];
    18	    private locked = false;
    19	
    20	    async lock(queueTimeoutMs: number = 60_000): Promise<() => void> {
    21	        return new Promise((resolve, reject) => {
    22	            let settled = false;
    23	            let timer: NodeJS.Timeout;
    24	            const release = () => {
    25	                if (this.queue.length > 0) this.queue.shift()!();
    26	                else this.locked = false;
    27	            };
    28	            const doResolve = () => {
    29	                if (!settled) {
    30	                    settled = true;
    31	                    clearTimeout(timer);
    32	                    resolve(release);
    33	                }
    34	            };
    35	            if (!this.locked) {
    36	                this.locked = true;
    37	                doResolve();
    38	            } else {
    39	                timer = setTimeout(() => {
    40	                    if (!settled) {
    41	                        settled = true;
    42	                        const idx = this.queue.indexOf(doResolve);
    43	                        if (idx !== -1) this.queue.splice(idx, 1);
    44	                        reject(new Error(`[NREKI] Mutex queue timeout after ${queueTimeoutMs}ms - deadlock prevented`));
    45	                    }
    46	                }, queueTimeoutMs);
    47	                this.queue.push(doResolve);
    48	            }
    49	        });
    50	    }
    51	
    52	    async withLock<T>(fn: () => Promise<T> | T): Promise<T> {
    53	        const unlock = await this.lock();
    54	        try {
    55	            return await fn();
    56	        } finally {
    57	            unlock();
    58	        }
    59	    }
    60	}

   174	     * Convierte una posición LSP (Línea, Carácter 0-indexed)

   177	    private getLspOffset(content: string, line: number, character: number): number {
   178	        let currentLine = 0;
   179	        let offset = 0;
   180	
   181	        while (currentLine < line && offset < content.length) {
   182	            const nl = content.indexOf('\n', offset);
   183	            if (nl === -1) break;
   184	            offset = nl + 1;
   185	            currentLine++;
   186	        }
   187	
   188	        return Math.min(offset + character, content.length);
   189	    }

   191	    /**
   192	     * TTRD Sintáctico - Micro-Scanner Híbrido (O(N), Cero dependencias).
   193	     * Extrae firmas de Python y Go crudas del VFS a prueba de
   194	     * formateadores (Black/Ruff/gofmt) y strings embebidos.
   195	     * Usa Regex SOLO para anclar el inicio (def/func), y un
   196	     * Bracket Balancer para parsear el interior.
   197	     */
   198	    /**
   199	     * TTRD Sintáctico v2.1 — Hardened Regex Scanner.
   200	     * Fixes over v2:
   201	     *   - Python: ^\s* instead of ^ to capture indented class methods
   202	     *   - Python: Triple-quoted string ("""/''') skip before bracket balancer
   203	     *   - Go: [a-zA-Z_] instead of [A-Z] to capture private functions
   204	     */
   205	    private extractRawSignatures(content: string, ext: string): Map<string, string> {
   206	        const signatures = new Map<string, string>();
   207	        if (ext !== ".py" && ext !== ".go") return signatures;
   208	
   209	        // FIX: ^\s* captures indented class methods. [a-zA-Z_] captures Go private funcs.
   210	        const startRegex = ext === ".py"
   211	            ? /^\s*(?:async\s+)?def\s+([a-zA-Z_]\w*)\s*\(/gm
   212	            : /^func\s+(?:\([^)]*\)\s+)?([a-zA-Z_][a-zA-Z0-9_]*)\s*\(/gm;
   213	
   214	        let match;
   215	        while ((match = startRegex.exec(content)) !== null) {
   216	            const symbolName = match[1];
   217	            const startIndex = match.index;
   218	            let i = startIndex + match[0].length - 1;
   219	
   220	            let parenDepth = 0;
   221	            let inString: string | null = null;
   222	            let foundEnd = false;
   223	
   224	            for (; i < content.length; i++) {
   225	                const ch = content[i];
   226	                if (inString) {
   227	                    if (ch === '\\') i++;
   228	                    else if (ch === inString) inString = null;
   229	                    continue;
   230	                }
   231	                // FIX: Triple-quoted string detection (Python """/''')
   232	                if ((ch === '"' || ch === "'") && i + 2 < content.length
   233	                    && content[i + 1] === ch && content[i + 2] === ch) {
   234	                    const tripleQuote = ch + ch + ch;
   235	                    const endTriple = content.indexOf(tripleQuote, i + 3);
   236	                    if (endTriple !== -1) { i = endTriple + 2; continue; }
   237	                    break; // Unterminated triple-quote — abort this signature
   238	                }
   239	                if (ch === '"' || ch === "'" || ch === '`') {
   240	                    inString = ch; continue;
   241	                }
   242	                if (ch === '(') parenDepth++;
   243	                else if (ch === ')') {
   244	                    parenDepth--;
   245	                    if (parenDepth === 0) { foundEnd = true; i++; break; }
   246	                }
   247	            }
   248	
   249	            if (foundEnd) {
   250	                let sigEnd = i;
   251	                if (ext === ".py") {
   252	                    while (sigEnd < content.length && content[sigEnd] !== ':') sigEnd++;
   253	                    if (sigEnd < content.length) sigEnd++;
   254	                } else if (ext === ".go") {
   255	                    while (sigEnd < content.length && content[sigEnd] !== '{') sigEnd++;
   256	                }
   257	                const rawSig = content.substring(startIndex, sigEnd).replace(/\s+/g, ' ').trim();
   258	                signatures.set(symbolName, rawSig);
   259	            }
   260	        }
   261	        return signatures;
   262	    }

   264	    /**
   265	     * TTRD Sintáctico - Detector de Regresión por Toxicidad Estructural.
   266	     * NO cuenta caracteres. Detecta 3 crímenes específicos:
   267	     * 1. Inyección tóxica (Any/interface{} donde no existía)
   268	     * 2. Colapso de retorno (pérdida de -> en Python, return type en Go)
   269	     * 3. Amnesia de parámetros (pérdida total de anotaciones de tipo)
   270	     */
   271	    private detectSignatureRegression(
   272	        oldSig: string,
   273	        newSig: string,
   274	        ext: string
   275	    ): { isRegression: boolean; reason: string } {
   276	        // 1. INYECCIÓN TÓXICA
   277	        const toxicRegex = ext === ".py" ? /\b(Any|any)\b/ : /\b(any|interface\{\})\b/;
   278	        if (!toxicRegex.test(oldSig) && toxicRegex.test(newSig)) {
   279	            return { isRegression: true, reason: `Injected toxic untyped '${ext === ".py" ? "Any" : "interface{}"}'` };
   280	        }
   281	
   282	        // 2. COLAPSO ESTRUCTURAL (Pérdida del Retorno)
   283	        if (ext === ".py") {
   284	            const hadReturn = /->\s*[^:]+/.test(oldSig);
   285	            const hasReturn = /->\s*[^:]+/.test(newSig);
   286	            if (hadReturn && !hasReturn) {
   287	                return { isRegression: true, reason: "Lost return type annotation (->)" };
   288	            }
   289	
   290	            // 3. AMNESIA DE PARÁMETROS (Pérdida total)
   291	            const oldParams = oldSig.match(/\((.*)\)/)?.[1] || "";
   292	            const newParams = newSig.match(/\((.*)\)/)?.[1] || "";
   293	            const oldAnnotations = (oldParams.match(/:/g) || []).length;
   294	            const newAnnotations = (newParams.match(/:/g) || []).length;
   295	
   296	            if (oldAnnotations > 0 && newAnnotations === 0) {
   297	                return { isRegression: true, reason: "Stripped all parameter type annotations" };
   298	            }
   299	        } else if (ext === ".go") {
   300	            const oldReturn = oldSig.substring(oldSig.lastIndexOf(')') + 1).replace('{', '').trim();
   301	            const newReturn = newSig.substring(newSig.lastIndexOf(')') + 1).replace('{', '').trim();
   302	            if (oldReturn.length > 0 && newReturn.length === 0) {
   303	                return { isRegression: true, reason: "Lost return type annotation" };
   304	            }
   305	        }
   306	
   307	        return { isRegression: false, reason: "" };
   308	    }

   320	    /** Validate cache against current TS version. Returns true if cache is usable. */
   321	    private validateBuildInfoCache(): boolean {
   322	        const versionFile = path.join(path.dirname(this.tsBuildInfoPath), "ts-version");
   323	        try {
   324	            if (!fs.existsSync(this.tsBuildInfoPath)) return false;
   325	            if (!fs.existsSync(versionFile)) {
   326	                fs.unlinkSync(this.tsBuildInfoPath);
   327	                return false;
   328	            }
   329	            const cached = fs.readFileSync(versionFile, "utf-8").trim();
   330	            if (cached !== ts.version) {
   331	                fs.unlinkSync(this.tsBuildInfoPath);
   332	                logger.warn(`TS version changed (${cached} -> ${ts.version}). Cache purged.`);
   333	                return false;
   334	            }
   335	            return true;
   336	        } catch {
   337	            return false; // First boot or corrupted cache
   338	        }
   339	    }

   351	    /**
   352	     * Boot the kernel. Applies CompilerHost overrides for VFS I/O.
   353	     * Cost: ~5-10s for initial semantic graph. Runs in background.
   354	     */
   355	    public boot(workspacePath: string, mode?: "file" | "project" | "hologram"): void {
   356	        if (this.booted) throw new Error("[NREKI] Kernel already booted");
   357	        if (mode) this.mode = mode;
   358	        logger.info(`Booting VFS-LSP Kernel (${this.mode} mode). Applying CompilerHost overrides...`);
   359	        this.projectRoot = this.toPosix(path.resolve(workspacePath));
   360	        this.initConfig();
   361	
   362	        // Hologram mode: override rootNames and compilerOptions
   363	        if (this.mode === "hologram") {
   364	            if (this.hasJitHologram() && this.prunedFiles.size === 0) {
   365	                // JIT mode: no pre-computed shadows. Keep only .d.ts/.d.mts/.d.cts from rootNames.
   366	                this.jitMode = true;
   367	                this.rootNames = new Set(
   368	                    [...this.rootNames].filter(f => /\.d\.[mc]?ts$/i.test(f)),
   369	                );
   370	                logger.info(`JIT Holography active. rootNames: ${this.rootNames.size} (.d.ts only). Shadows on-demand.`);
   371	            } else {
   372	                // Eager mode: pre-computed shadows available
   373	                this.jitMode = false;
   374	                const ambientSet = new Set(
   375	                    this.ambientFiles.map(f => this.toPosix(path.resolve(this.projectRoot, f))),
   376	                );
   377	                this.rootNames = new Set(
   378	                    [...this.rootNames].filter(f => ambientSet.has(f)),
   379	                );
   380	            }
   381	            // CompilerOptions: always set for hologram
   382	            this.compilerOptions.noEmit = false;
   383	            this.compilerOptions.declaration = true;
   384	            this.compilerOptions.emitDeclarationOnly = true;
   385	            this.compilerOptions.isolatedModules = false;
   386	            // Sync filtered rootNames back to backend (hologram creates new Set)
   387	            this.tsBackend.rootNames = this.rootNames;
   388	        }
   389	
   390	        // ─── Delegate CompilerHost creation to backend (Strangler Fig 2B) ───
   391	        // The kernel creates VfsAdapter closures over its own state.
   392	        // The backend uses them to create CompilerHost + LanguageService.
   393	        // VFS, hologram, JIT state stay in the kernel. Always.
   394	        const self = this;
   395	        const vfsAdapter: import("./backends/ts-compiler-wrapper.js").VfsAdapter = {
   396	            readFile(fileName: string): string | undefined {
   397	                const posixPath = self.toPosix(path.resolve(self.projectRoot, fileName));
   398	
   399	                // HOLOGRAM INTERCEPT: serve shadow .d.ts or hide pruned .ts
   400	                if (self.mode === "hologram") {
   401	                    if (/\.d\.[mc]?ts$/i.test(fileName)) {
   402	                        const tsPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1ts");
   403	                        const shadow = self.shadowContent.get(tsPath);
   404	                        if (shadow !== undefined) return shadow;
   405	                        const tsxPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1tsx");
   406	                        const shadow2 = self.shadowContent.get(tsxPath);
   407	                        if (shadow2 !== undefined) return shadow2;
   408	                        if (self.jitMode) {
   409	                            if (!self.jitClassifiedCache.has(tsPath)) {
   410	                                self.jitClassifyFile(tsPath);
   411	                                const shadowJ = self.shadowContent.get(tsPath);
   412	                                if (shadowJ !== undefined) return shadowJ;
   413	                            }
   414	                            if (!self.jitClassifiedCache.has(tsxPath)) {
   415	                                self.jitClassifyFile(tsxPath);
   416	                                const shadowJ2 = self.shadowContent.get(tsxPath);
   417	                                if (shadowJ2 !== undefined) return shadowJ2;
   418	                            }
   419	                        }
   420	                    }
   421	                    if (self.prunedTsLookup.has(posixPath) && !self.currentEditTargets.has(posixPath)) {
   422	                        return undefined;
   423	                    }
   424	                }
   425	
   426	                if (self.vfs.has(posixPath)) {
   427	                    const content = self.vfs.get(posixPath);
   428	                    return content === null ? undefined : content;
   429	                }
   430	                if (isSensitivePath(fileName)) {
   431	                    return undefined;
   432	                }
   433	                // Fallback to disk via backend's original readFile
   434	                return self.tsBackend.originalReadFile.call(self.tsBackend.host, fileName);
   435	            },
   436	
   437	            fileExists(fileName: string): boolean {
   438	                const posixPath = self.toPosix(path.resolve(self.projectRoot, fileName));
   439	
   440	                if (self.mode === "hologram") {
   441	                    if (self.currentEditTargets.has(posixPath)) {
   442	                        return self.tsBackend.originalFileExists.call(self.tsBackend.host, fileName);
   443	                    }
   444	                    if (self.prunedTsLookup.has(posixPath)) return false;
   445	                    if (self.shadowDtsLookup.has(posixPath)) return true;
   446	
   447	                    if (self.jitMode) {
   448	                        if (/\.tsx?$/.test(fileName) && !/\.d\.[mc]?ts$/i.test(fileName)) {
   449	                            if (self.tsBackend.originalFileExists.call(self.tsBackend.host, fileName)) {
   450	                                self.jitClassifyFile(posixPath);
   451	                                if (self.prunedTsLookup.has(posixPath)) return false;
   452	                            }
   453	                        }
   454	                        if (/\.d\.[mc]?ts$/i.test(fileName)) {
   455	                            const tsPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1ts");
   456	                            const tsxPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1tsx");
   457	                            if (!self.jitClassifiedCache.has(tsPath) && self.tsBackend.originalFileExists.call(self.tsBackend.host, tsPath)) {
   458	                                if (self.jitClassifyFile(tsPath)) return true;
   459	                            }
   460	                            if (!self.jitClassifiedCache.has(tsxPath) && self.tsBackend.originalFileExists.call(self.tsBackend.host, tsxPath)) {
   461	                                if (self.jitClassifyFile(tsxPath)) return true;
   462	                            }
   463	                        }
   464	                    }
   465	                }
   466	
   467	                if (self.vfs.has(posixPath)) return self.vfs.get(posixPath) !== null;
   468	                return self.tsBackend.originalFileExists.call(self.tsBackend.host, fileName);
   469	            },
   470	
   471	            getModifiedTime(fileName: string): Date {
   472	                const posixPath = self.toPosix(path.resolve(self.projectRoot, fileName));
   473	                if (self.vfsClock.has(posixPath)) return self.vfsClock.get(posixPath)!;
   474	                return ts.sys.getModifiedTime?.(fileName) ?? new Date();
   475	            },
   476	
   477	            directoryExists(dirName: string): boolean {
   478	                const posixDir = self.toPosix(path.resolve(self.projectRoot, dirName));
   479	                if (ts.sys.directoryExists(dirName)) return true;
   480	                return self.vfsDirectories.has(posixDir);
   481	            },
   482	
   483	            getScriptVersion(fileName: string): string {
   484	                const posixPath = self.toPosix(path.resolve(self.projectRoot, fileName));
   485	                return self.vfsClock.get(posixPath)?.getTime().toString() || "1";
   486	            },
   487	
   488	            getScriptSnapshot(fileName: string): ts.IScriptSnapshot | undefined {
   489	                const posixPath = self.toPosix(path.resolve(self.projectRoot, fileName));
   490	
   491	                // HOLOGRAM INTERCEPT for LanguageService
   492	                if (self.mode === "hologram" && /\.d\.[mc]?ts$/i.test(fileName)) {
   493	                    const tsPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1ts");
   494	                    const shadow = self.shadowContent.get(tsPath);
   495	                    if (shadow !== undefined) return ts.ScriptSnapshot.fromString(shadow);
   496	                    const tsxPath = posixPath.replace(/\.d\.([mc]?)ts$/i, ".$1tsx");
   497	                    const shadow2 = self.shadowContent.get(tsxPath);
   498	                    if (shadow2 !== undefined) return ts.ScriptSnapshot.fromString(shadow2);
   499	                }
   500	
   501	                if (self.vfs.has(posixPath)) {
   502	                    const content = self.vfs.get(posixPath);
   503	                    if (content === null || content === undefined) return undefined;
   504	                    return ts.ScriptSnapshot.fromString(content);
   505	                }
   506	                if (!self.tsBackend.originalFileExists.call(self.tsBackend.host, fileName)) return undefined;
   507	                const content = self.tsBackend.originalReadFile.call(self.tsBackend.host, fileName);
   508	                if (!content) return undefined;
   509	                return ts.ScriptSnapshot.fromString(content);
   510	            },
   511	        };
   512	
   513	        this.tsBackend.createCompilerInfra(this.projectRoot, vfsAdapter);
   514	
   515	        // Sync host reference for test access (JIT holography tests)
   516	        this.host = this.tsBackend.host;
   517	        // ──────────────────────────────────────────────────────────────
   518	
   519	        // Incremental cache: delegate to backend
   520	        if (this.validateBuildInfoCache()) {
   521	            try {
   522	                const readBuildHost: ts.ReadBuildProgramHost = {
   523	                    useCaseSensitiveFileNames: () => ts.sys.useCaseSensitiveFileNames,
   524	                    getCurrentDirectory: () => this.projectRoot,
   525	                    readFile: (fileName: string) => {
   526	                        if (fileName.endsWith(".tsbuildinfo")) {
   527	                            return ts.sys.readFile(fileName);
   528	                        }
   529	                        return this.tsBackend.host.readFile(fileName);
   530	                    },
   531	                };
   532	                const oldProgram = ts.readBuilderProgram(
   533	                    this.tsBackend.compilerOptions, readBuildHost
   534	                ) as ts.EmitAndSemanticDiagnosticsBuilderProgram | undefined;
   535	                if (oldProgram) {
   536	                    this.tsBackend._injectBuilderProgram(oldProgram);
   537	                }
   538	            } catch {
   539	                // Corrupted cache, cold boot — backend starts fresh
   540	            }
   541	        }
   542	
   543	        this.tsBackend.updateProgram();
   544	        // Hologram/file mode: skip boot baseline - JIT baseline in interceptAtomicBatch
   545	        // handles it scoped to target files.
   546	        if (this.mode === "project") {
   547	            this.tsBackend.captureBaseline(undefined, this.mode);
   548	        }
   549	        // D2: WAL-aware crash recovery.
   550	        // If a previous session crashed mid-commit, the WAL tells us
   551	        // exactly which files to restore from their .bak backups.
   552	        const txDir = path.join(this.projectRoot, ".nreki", "transactions");
   553	        const walPath = path.join(txDir, "wal.json");
   554	        if (fs.existsSync(walPath)) {
   555	            try {
   556	                const wal = JSON.parse(fs.readFileSync(walPath, "utf-8"));
   557	                if (wal.status === "pending" && Array.isArray(wal.files)) {
   558	                    logger.warn(`WAL recovery: restoring ${wal.files.length} file(s) from crashed transaction`);
   559	                    for (const entry of wal.files) {
   560	                        if (entry.backup && fs.existsSync(entry.backup)) {
   561	                            fs.renameSync(entry.backup, entry.target);
   562	                        } else if (!entry.backup && fs.existsSync(entry.target)) {
   563	                            // File was newly created in the crashed tx — remove it
   564	                            fs.unlinkSync(entry.target);
   565	                        }
   566	                    }
   567	                }
   568	            } catch (walErr) {
   569	                logger.error(`WAL recovery failed: ${walErr}`);
   570	            }
   571	        }
   572	        if (fs.existsSync(txDir)) {
   573	            try { fs.rmSync(txDir, { recursive: true, force: true }); } catch { /* best effort */ }
   574	        }
   575	
   576	        // Capture initial error count at boot (immutable after this point)
   577	        this.bootErrorCount = this.getBaselineErrorCount();
   578	
   579	        this.booted = true;
   580	        // Log backend info for diagnostics
   581	        logger.info(`Backend: ${this.tsBackend.name} (healing: ${this.tsBackend.capabilities.supportsAutoHealing}, ttrd: ${this.tsBackend.capabilities.supportsTTRD})`);
   582	        logger.info(
   583	            `Kernel booted. Tracking ${this.rootNames.size} files. ` +
   584	            `Baseline: ${this.tsBackend.baselineCount} invariants. ` +
   585	            `Boot errors: ${this.bootErrorCount}.`
   586	        );
   587	    }

   590	        return this.booted;

   595	    /**
   596	     * Extract the compiler's resolved type for each locally-declared export.
   597	     * Uses TypeChecker to read resolved types, not AST text.

   602	    /**
   603	     * Check if a type string contains toxic type patterns.
   604	     * Uses word boundaries to avoid false positives on identifiers
   605	     * like "Company" or "ManyToMany".
   606	     *
   607	     * Shared with file fragility tracker via static method.
   608	     */
   609	    public static isToxicType(typeStr: string): boolean {
   610	        const s = typeStr.trim();
   611	        // Bare empty types
   612	        if (s === "object" || s.replace(/\s/g, "") === "{}") return true;
   613	        // Toxic keywords anywhere in the type string (handles parameters, returns, generics)
   614	        if (/\b(any|unknown|Function)\b/.test(s)) return true;
   615	        return false;
   616	    }

   622	     * Asymmetric weights:
   623	     *   any = 10 (disables all checking)
   624	     *   unknown = 2 (safe but unresolved)
   625	     *   Function = 5 (callable but untyped)
   626	     *
   627	     * Depth limit prevents CPU freeze on deeply nested types (Prisma, tRPC).
   628	     * Cycle-safe via Set<ts.Type> pointer comparison (TS interns types).

   632	    // ─── NREKI L3.3: Self-Healing Agent Loop ─────────────────────────
   633	
   634	    /**
   635	     * Iterative AST healing engine.
   636	     * Repairs structural errors one by one, incrementally recompiling
   637	     * the universe in RAM between each fix to ensure perfect byte offsets.
   638	     * If a fix causes cascade, performs micro-rollback and discards it.
   639	     *
   640	     * Strict Entropy Reduction Invariant:
   641	     *   |E(S₀ + Δ)| ≤ |E₀| - 1
   642	     *   Any patch resulting in ≥ |E₀| errors is reverted in O(1).
   643	     *
   644	     * Fixed bugs:
   645	     *   BUG 1: String(diag.code) → safe Number cast (TS expects number[])
   646	     *   BUG 2: explicitlyEditedFiles mutated by reference → cloned on entry
   647	     *   BUG 3: getSemanticDiagnosticsOfNextAffectedFile is stateful → compute ONCE
   648	     *   BUG 4: new Map(this.vfs) copied entire VFS → micro-UndoLog O(touched files)
   649	     *   BUG 5: localEditedFiles.delete(p) in micro-rollback blinded Shield 2 →
   650	     *          only delete files the healer added, never the LLM's files
   651	     */
   652	    private async attemptAutoHealing(
   653	        initialErrors: NrekiStructuredError[],
   654	        parentEditedFiles: Set<string>,
   655	        filesToEvaluate: Set<string>
   656	    ): Promise<{ healed: boolean; appliedFixes: string[]; newlyTouchedFiles: Set<string>; finalErrors: NrekiStructuredError[] }> {
   657	        if (initialErrors.length === 0) {
   658	            return { healed: false, appliedFixes: [], newlyTouchedFiles: new Set(), finalErrors: initialErrors };
   659	        }
   660	
   661	        const MAX_ITERATIONS = 10;
   662	        const fixDescriptions = new Set<string>();
   663	
   664	        const localEditedFiles = new Set(parentEditedFiles);
   665	        const newlyTouchedFiles = new Set<string>();
   666	
   667	        const healUndoLog = new Map<string, {
   668	            content: string | null | undefined; time: Date | undefined; wasInRoot: boolean;
   669	        }>();
   670	
   671	        const failedFixHashes = new Set<string>();
   672	
   673	        let currentErrors: NrekiStructuredError[] = initialErrors;
   674	        let iteration = 0;
   675	
   676	        while (currentErrors.length > 0 && iteration < MAX_ITERATIONS) {
   677	            let appliedAnyFix = false;
   678	
   679	            for (const error of currentErrors) {
   680	                const fixes = await this.tsBackend.getAutoFixes(error.file, error);
   681	                if (fixes.length === 0) continue;
   682	                const bestFix = fixes[0];
   683	                const fixHash = `${error.file}:${error.line}:${bestFix.description}`;
   684	                if (failedFixHashes.has(fixHash)) continue;
   685	
   686	                // Backup state before applying fix
   687	                const microUndoLog = new Map<string, {
   688	                    content: string | null | undefined; time: Date | undefined; wasInRoot: boolean;
   689	                }>();
   690	
   691	                for (const change of bestFix.changes) {
   692	                    const changePath = change.filePath;
   693	                    const state = {
   694	                        content: this.vfs.has(changePath) ? this.vfs.get(changePath) : undefined,
   695	                        time: this.vfsClock.get(changePath),
   696	                        wasInRoot: this.rootNames.has(changePath),
   697	                    };
   698	                    microUndoLog.set(changePath, state);
   699	                    if (!healUndoLog.has(changePath)) healUndoLog.set(changePath, state);
   700	                }
   701	
   702	                // Apply fix to VFS
   703	                for (const change of bestFix.changes) {
   704	                    const changePath = change.filePath;
   705	                    let content = this.vfs.get(changePath) ?? this.tsBackend.host.readFile(changePath) ?? "";
   706	
   707	                    const sortedChanges = [...change.textChanges].sort((a, b) => b.start - a.start);
   708	
   709	                    for (const textChange of sortedChanges) {
   710	                        const start = textChange.start;
   711	                        const end = start + textChange.length;
   712	                        content = content.slice(0, start) + textChange.newText + content.slice(end);
   713	                    }
   714	
   715	                    this.vfs.set(changePath, content);
   716	                    this.vfsClock.set(changePath, new Date(this.logicalTime));
   717	                    localEditedFiles.add(changePath);
   718	
   719	                    if (!parentEditedFiles.has(changePath)) newlyTouchedFiles.add(changePath);
   720	                    if (!this.rootNames.has(changePath) && this.isTypeScriptFile(changePath)) {
   721	                        this.rootNames.add(changePath);
   722	                    }
   723	                }
   724	
   725	                // Recompile after fix
   726	                this.logicalTime += 1000;
   727	                this.tsBackend.updateProgram();
   728	                const newEvaluate = new Set(filesToEvaluate);
   729	                for (const f of localEditedFiles) newEvaluate.add(f);
   730	                const newErrors = await this.tsBackend.getDiagnostics(newEvaluate, localEditedFiles, this.mode);
   731	
   732	                if (newErrors.length >= currentErrors.length) {
   733	                    // MICRO-ROLLBACK
   734	                    for (const [p, state] of microUndoLog.entries()) {
   735	                        if (state.content !== undefined) this.vfs.set(p, state.content); else this.vfs.delete(p);
   736	                        if (state.time) this.vfsClock.set(p, state.time); else this.vfsClock.delete(p);
   737	                        if (state.wasInRoot) this.rootNames.add(p); else this.rootNames.delete(p);
   738	
   739	                        if (!parentEditedFiles.has(p)) {
   740	                            localEditedFiles.delete(p);
   741	                        }
   742	                        newlyTouchedFiles.delete(p);
   743	                    }
   744	                    this.logicalTime += 1000;
   745	                    this.tsBackend.updateProgram();
   746	                    failedFixHashes.add(fixHash);
   747	                } else {
   748	                    // FIX ACCEPTED
   749	                    const changesPreview = bestFix.changes.map(c => {
   750	                        const file = path.basename(c.filePath);
   751	                        const texts = c.textChanges.map(t => t.newText.trim()).filter(Boolean);
   752	                        return `${file}: [${texts.join(", ")}]`;
   753	                    }).join(" | ");
   754	
   755	                    fixDescriptions.add(`- ${bestFix.description} -> \`${changesPreview}\``);
   756	                    appliedAnyFix = true;
   757	                    currentErrors = newErrors;
   758	
   759	                    break;
   760	                }
   761	            }
   762	
   763	            if (!appliedAnyFix) break;
   764	            iteration++;
   765	        }
   766	
   767	        // Healing result evaluation
   768	        if (currentErrors.length > 0) {
   769	            if (healUndoLog.size > 0) {
   770	                for (const [p, state] of healUndoLog.entries()) {
   771	                    if (state.content !== undefined) this.vfs.set(p, state.content); else this.vfs.delete(p);
   772	                    if (state.time) this.vfsClock.set(p, state.time); else this.vfsClock.delete(p);
   773	                    if (state.wasInRoot) this.rootNames.add(p); else this.rootNames.delete(p);
   774	                }
   775	                this.logicalTime += 1000;
   776	                for (const p of healUndoLog.keys()) {
   777	                    if (this.vfs.has(p)) this.vfsClock.set(p, new Date(this.logicalTime));
   778	                }
   779	                this.tsBackend.updateProgram();
   780	            }
   781	
   782	            this._healingStats.failed++;
   783	            return { healed: false, appliedFixes: [], newlyTouchedFiles: new Set(), finalErrors: initialErrors };
   784	        }
   785	
   786	        this._healingStats.applied++;
   787	        for (const file of localEditedFiles) parentEditedFiles.add(file);
   788	
   789	        return {
   790	            healed: true,
   791	            appliedFixes: Array.from(fixDescriptions),
   792	            newlyTouchedFiles,
   793	            finalErrors: []
   794	        };
   795	    }
   796	
   797	    /**
   798	     * NREKI L3.4: LSP Auto-Healing (Go / Python)
   799	     * DISEÑO ACORAZADO:
   800	     * - Latencia JSON-RPC es alta (~300ms/ciclo). Límite estricto de 2 iteraciones.
   801	     * - Whitelist ultra-conservadora: SOLO CodeActions con "import" o "add" en título.
   802	     * - Rechaza explícitamente "remove" y "delete" (destructivos).
   803	     * - Micro-rollback dual: VFS de NREKI + VFS del sidecar (cura Split-Brain).
   804	     * - Macro-rollback completo si quedan errores al final.
   805	     */
   806	    private async attemptLspAutoHealing(
   807	        sidecar: LspSidecarBase,
   808	        initialErrors: NrekiStructuredError[],
   809	        parentEditedFiles: Set<string>
   810	    ): Promise<{ healed: boolean; appliedFixes: string[]; newlyTouchedFiles: Set<string>; finalErrors: NrekiStructuredError[] }> {
   811	
   812	        if (initialErrors.length === 0 || sidecar.isDead) {
   813	            return { healed: false, appliedFixes: [], newlyTouchedFiles: new Set(), finalErrors: initialErrors };
   814	        }
   815	
   816	        const MAX_ITERATIONS = 2;
   817	        const fixDescriptions = new Set<string>();
   818	        const localEditedFiles = new Set(parentEditedFiles);
   819	        const newlyTouchedFiles = new Set<string>();
   820	
   821	        const healUndoLog = new Map<string, { content: string | null | undefined; time: Date | undefined }>();
   822	        const failedFixHashes = new Set<string>();
   823	
   824	        let currentErrors = initialErrors;
   825	        let iteration = 0;
   826	
   827	        while (currentErrors.length > 0 && iteration < MAX_ITERATIONS) {
   828	            let appliedAnyFix = false;
   829	
   830	            for (const error of currentErrors) {
   831	                const lspRange = {
   832	                    start: { line: error.line - 1, character: Math.max(0, error.column - 1) },
   833	                    end: { line: error.line - 1, character: error.column }
   834	                };
   835	
   836	                const diagnostic = {
   837	                    range: lspRange,
   838	                    message: error.message,
   839	                    code: error.code.replace(`${sidecar.languageId}-`, ""),
   840	                };
   841	
   842	                let fixes: any[] = [];
   843	                try {
   844	                    fixes = await (sidecar as any).requestCodeActions(error.file, diagnostic);
   845	                } catch { continue; }
   846	
   847	                if (!fixes || fixes.length === 0) continue;
   848	
   849	                // EL MURO DE HIELO: Whitelist estricta por title
   850	                const safeFixes = fixes.filter((f: any) => {
   851	                    const title = (f.title || "").toLowerCase();
   852	                    if (title.includes("remove") || title.includes("delete")) return false;
   853	                    return title.includes("import") || title.includes("add ");
   854	                });
   855	
   856	                if (safeFixes.length === 0) continue;
   857	
   858	                const bestFix = safeFixes[0];
   859	                const fixDesc = bestFix.title || `Add import in ${path.basename(error.file)}`;
   860	                const fixHash = `${error.file}:${error.line}:${fixDesc}`;
   861	
   862	                if (failedFixHashes.has(fixHash)) continue;
   863	
   864	                const changePath = this.toPosix(path.resolve(this.projectRoot, bestFix.filePath || error.file));
   865	
   866	                // Backup VFS (Micro-Rollback log)
   867	                const state = {
   868	                    content: this.vfs.has(changePath) ? this.vfs.get(changePath) : undefined,
   869	                    time: this.vfsClock.get(changePath)
   870	                };
   871	                if (!healUndoLog.has(changePath)) healUndoLog.set(changePath, state);
   872	
   873	                // Aplicar Fix traduciendo coordenadas LSP → byte offsets
   874	                let content = this.vfs.get(changePath) ?? fs.readFileSync(changePath, "utf-8");
   875	
   876	                const startIdx = this.getLspOffset(content, bestFix.range.start.line, bestFix.range.start.character);
   877	                const endIdx = this.getLspOffset(content, bestFix.range.end.line, bestFix.range.end.character);
   878	
   879	                content = content.slice(0, startIdx) + bestFix.newText + content.slice(endIdx);
   880	
   881	                this.vfs.set(changePath, content);
   882	                this.vfsClock.set(changePath, new Date(this.logicalTime));
   883	                localEditedFiles.add(changePath);
   884	                if (!parentEditedFiles.has(changePath)) newlyTouchedFiles.add(changePath);
   885	
   886	                // Re-evaluar contra el Sidecar (Round-Trip)
   887	                this.logicalTime += 1000;
   888	                let newErrors: NrekiStructuredError[] = [];
   889	                try {
   890	                    const scEdits = Array.from(localEditedFiles).map(f => ({
   891	                        filePath: f, content: this.vfs.get(f) ?? null
   892	                    }));
   893	                    newErrors = await sidecar.validateEdits(scEdits);
   894	                } catch {
   895	                    newErrors = currentErrors;
   896	                }
   897	
   898	                // TEOREMA DE REDUCCIÓN ESTRICTA — GLOBAL (not per-file)
   899	                // Evaluate against the ENTIRE error pool. A fix that silences
   900	                // api.py but breaks 50 consumers must be rejected.
   901	                if (newErrors.length >= currentErrors.length) {
   902	                    // MICRO-ROLLBACK NREKI VFS
   903	                    if (state.content !== undefined) this.vfs.set(changePath, state.content);
   904	                    else this.vfs.delete(changePath);
   905	
   906	                    if (state.time) this.vfsClock.set(changePath, state.time);
   907	                    else this.vfsClock.delete(changePath);
   908	
   909	                    if (!parentEditedFiles.has(changePath)) localEditedFiles.delete(changePath);
   910	                    newlyTouchedFiles.delete(changePath);
   911	
   912	                    // MICRO-ROLLBACK SIDECAR VFS (Synchronous — cures Split-Brain)
   913	                    try {
   914	                        await sidecar.validateEdits([{ filePath: changePath, content: state.content ?? null }]);
   915	                    } catch {
   916	                        sidecar.isDead = true;
   917	                    }
   918	
   919	                    failedFixHashes.add(fixHash);
   920	                } else {
   921	                    // FIX ACEPTADO
   922	                    fixDescriptions.add(`- LSP Auto-Heal (${sidecar.languageId}): ${fixDesc}`);
   923	                    appliedAnyFix = true;
   924	                    currentErrors = newErrors;
   925	                    break;
   926	                }
   927	            }
   928	
   929	            if (!appliedAnyFix) break;
   930	            iteration++;
   931	        }
   932	
   933	        // MACRO-ROLLBACK si quedan errores
   934	        if (currentErrors.length > 0) {
   935	            const rollbacks = [];
   936	            for (const [p, state] of healUndoLog.entries()) {
   937	                if (state.content !== undefined) this.vfs.set(p, state.content);
   938	                else this.vfs.delete(p);
   939	                rollbacks.push({ filePath: p, content: state.content ?? null });
   940	            }
   941	            if (rollbacks.length > 0) {
   942	                this.logicalTime += 1000;
   943	                try {
   944	                    await sidecar.validateEdits(rollbacks);
   945	                } catch {
   946	                    sidecar.isDead = true;
   947	                }
   948	            }
   949	            this._healingStats.failed++;
   950	            return { healed: false, appliedFixes: [], newlyTouchedFiles: new Set(), finalErrors: initialErrors };
   951	        }
   952	
   953	        this._healingStats.applied++;
   954	        for (const file of localEditedFiles) parentEditedFiles.add(file);
   955	
   956	        return { healed: true, appliedFixes: Array.from(fixDescriptions), newlyTouchedFiles, finalErrors: [] };
   957	    }
   958	
   959	    /**
   960	     * SINGLE ENTRY POINT (P21, P22).
   961	     * Atomic batch validation: inject all edits into VFS, evaluate macro-state.
   962	     * Triple shield: Global → Syntactic → Semantic.
   963	     */
   964	    public async interceptAtomicBatch(edits: NrekiEdit[], dependents: string[] = []): Promise<NrekiInterceptResult> {
   965	        if (!this.booted) throw new Error("[NREKI] Kernel not booted");
   966	        if (!edits || edits.length === 0) return { safe: true, exitCode: 0, latencyMs: "0.00" };
   967	
   968	        // Corruption guard: if a previous timeout left the VFS in a partial state, rebuild
   969	        if (this.tsBackend.isCorrupted) {
   970	            this.tsBackend.purgeCache();
   971	            logger.warn("Rebuilding after timeout-corrupted state.");
   972	        }
   973	
   974	        return this.mutex.withLock(async () => {
   975	        const txId = crypto.randomBytes(4).toString("hex");
   976	        setTxId(txId);
   977	        logger.info(`interceptAtomicBatch: ${edits.length} file(s)`);
   978	        try {
   979	        const t0 = performance.now();
   980	
   981	        const rollbackState = new Map<string, {
   982	            content: string | null | undefined;
   983	            time: Date | undefined;
   984	            wasInRoot: boolean;
   985	        }>();
   986	
   987	            // B6: Save logicalTime for rollback
