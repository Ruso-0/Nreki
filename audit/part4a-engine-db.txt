   211	    // ─── File Indexing ─────────────────────────────────────────────
   212	
   213	    /**
   214	     * Index a single file: parse AST → generate embeddings → store chunks.
   215	     * Skips files whose SHA-256 hash hasn't changed (Merkle diffing).
   216	     *
   217	     * In Lite mode, stores chunks without embeddings (keyword search only).
   218	     * In Pro mode, generates embeddings for hybrid search.
   219	     */
   220	    async indexFile(filePath: string): Promise<ParseResult | null> {
   221	        if (this.config.enableEmbeddings) {
   222	            await this.initializeEmbedder();
   223	        } else {
   224	            await this.initialize();
   225	        }
   226	
   227	        // FIX 7: Check file size and extension before processing
   228	        let stat: fs.Stats;
   229	        try {
   230	            stat = fs.statSync(filePath);
   231	        } catch {
   232	            return null;
   233	        }
   234	
   235	        const filterResult = shouldProcess(filePath, stat.size);
   236	        if (!filterResult.process) {
   237	            return null;
   238	        }
   239	
   240	        // Read file content (BOM-safe for tree-sitter)
   241	        let content: string;
   242	        try {
   243	            content = readSource(filePath);
   244	        } catch {
   245	            return null; // File may have been deleted
   246	        }
   247	
   248	        // Check extension support - unsupported files get plaintext fallback
   249	        const ext = path.extname(filePath).toLowerCase();
   250	        if (!this.parser.isSupported(ext)) {
   251	            // Plaintext fallback: index entire file as one chunk for BM25 keyword search.
   252	            // No AST features (no validation, no structural compression, no semantic edit)
   253	            // but search WILL find content in this file.
   254	            if (!this.db.fileNeedsUpdate(filePath, content)) return null;
   255	
   256	            const lineCount = content.split("\n").length;
   257	            // Use content as shorthand so BM25 keyword index can search it
   258	            const shorthand = `[file] ${path.basename(filePath)} (${lineCount} lines)\n${content}`;
   259	
   260	            this.db.clearChunks(filePath);
   261	
   262	            if (this.config.enableEmbeddings) {
   263	                await this.initializeEmbedder();
   264	                const { embedding } = await this.embedder.embed(content.slice(0, 1000));
   265	                this.db.insertChunk(filePath, shorthand, content, "file", 1, lineCount, embedding);
   266	            } else {
   267	                this.db.insertChunk(filePath, shorthand, content, "file", 1, lineCount, new Float32Array(0));
   268	            }
   269	
   270	            const hash = this.db.hashContent(content);
   271	            this.db.upsertFile(filePath, hash);
   272	
   273	            return {
   274	                filePath,
   275	                chunks: [{ shorthand, rawCode: content, nodeType: "file", startLine: 1, endLine: lineCount, startIndex: 0, endIndex: content.length, symbolName: "" }],
   276	                totalLines: lineCount,
   277	                language: ext.slice(1),
   278	            };
   279	        }
   280	
   281	        // Merkle diffing: skip unchanged files
   282	        if (!this.db.fileNeedsUpdate(filePath, content)) return null;
   283	
   284	        // Parse AST
   285	        const result = await this.parser.parse(filePath, content);
   286	        if (result.chunks.length === 0) return result;
   287	
   288	        // Clear old chunks for this file
   289	        this.db.clearChunks(filePath);
   290	
   291	        // Generate embeddings (or null placeholder) and store in batch
   292	        const chunkData: Array<{
   293	            path: string;
   294	            shorthand: string;
   295	            rawCode: string;
   296	            nodeType: string;
   297	            startLine: number;
   298	            endLine: number;
   299	            embedding: Float32Array;
   300	            startIndex: number;
   301	            endIndex: number;
   302	            symbolName: string;
   303	        }> = [];
   304	
   305	        for (const chunk of result.chunks) {
   306	            let embedding: Float32Array;
   307	            if (this.config.enableEmbeddings) {
   308	                ({ embedding } = await this.embedder.embed(chunk.shorthand));
   309	            } else {
   310	                // Lite mode: empty embedding placeholder
   311	                embedding = new Float32Array(0);
   312	            }
   313	            chunkData.push({
   314	                path: filePath,
   315	                shorthand: chunk.shorthand,
   316	                rawCode: chunk.rawCode,
   317	                nodeType: chunk.nodeType,
   318	                startLine: chunk.startLine,
   319	                endLine: chunk.endLine,
   320	                embedding,
   321	                startIndex: chunk.startIndex,
   322	                endIndex: chunk.endIndex,
   323	                symbolName: chunk.symbolName,
   324	            });
   325	        }
   326	
   327	        // Batch insert with transaction
   328	        this.db.insertChunksBatch(chunkData);
   329	
   330	        // Update file hash
   331	        const hash = this.db.hashContent(content);
   332	        this.db.upsertFile(filePath, hash);
   333	
   334	        return result;
   335	    }

   337	    /**
   338	     * Index an entire directory recursively.
   339	     * Walks all supported files and indexes them in sequence.
   340	     */
   341	    async indexDirectory(dirPath: string): Promise<{
   342	        indexed: number;
   343	        skipped: number;
   344	        errors: number;
   345	    }> {
   346	        if (this.config.enableEmbeddings) {
   347	            await this.initializeEmbedder();
   348	        } else {
   349	            await this.initialize();
   350	        }
   351	
   352	        let indexed = 0;
   353	        let skipped = 0;
   354	        let errors = 0;
   355	
   356	        const files = this.walkDirectory(dirPath);
   357	        let processedCount = 0;
   358	
   359	        for (const file of files) {
   360	            try {
   361	                const result = await this.indexFile(file);
   362	                if (result && result.chunks.length > 0) {
   363	                    indexed++;
   364	                } else {
   365	                    skipped++;
   366	                }
   367	            } catch (err) {
   368	                logger.error(
   369	                    `Error indexing ${file}: ${(err as Error).message}`
   370	                );
   371	                errors++;
   372	            }
   373	
   374	            // FIX 3: Yield event loop every 100 files to avoid blocking
   375	            processedCount++;
   376	            if (processedCount % 100 === 0) {
   377	                await new Promise<void>(resolve => setImmediate(resolve));
   378	            }
   379	        }
   380	
   381	        // Persist to disk after batch indexing
   382	        this.db.save();
   383	
   384	        return { indexed, skipped, errors };
   385	    }

   387	    /** Recursively walk a directory and return all supported files. */
   388	    private walkDirectory(dirPath: string): string[] {
   389	        const files: string[] = [];
   390	        const isIgnored = picomatch(this.config.ignorePaths);
   391	
   392	        const walk = (dir: string) => {
   393	            let entries: fs.Dirent[];
   394	            try {
   395	                entries = fs.readdirSync(dir, { withFileTypes: true });
   396	            } catch {
   397	                return;
   398	            }
   399	
   400	            for (const entry of entries) {
   401	                const fullPath = path.join(dir, entry.name);
   402	                const relativePath = path.relative(dirPath, fullPath).replace(/\\/g, "/");
   403	
   404	                if (entry.isDirectory()) {
   405	                    // Check directory name and relative path against glob patterns
   406	                    if (!isIgnored(entry.name) && !isIgnored(relativePath) && !isIgnored(relativePath + "/")) {
   407	                        walk(fullPath);
   408	                    }
   409	                    continue;
   410	                }
   411	
   412	                // Check file against ignore patterns, then check extension
   413	                if (isIgnored(relativePath) || isIgnored(entry.name)) continue;
   414	                const ext = path.extname(entry.name).toLowerCase();
   415	                if (this.config.extensions.includes(ext)) {
   416	                    files.push(fullPath);
   417	                }
   418	            }
   419	        };
   420	
   421	        walk(dirPath);
   422	        return files;
   423	    }

   427	    /**
   428	     * T-RAG (Topology-Aware RAG) - Blast Radius-Aware Search.
   429	     * Re-ranks standard semantic/keyword search results using the project's dependency graph.
   430	     * Injects PageRank (mass), InDegree (gravity), and Blast Radius awareness directly into the LLM's context.
   431	     *
   432	     * TRS (Tectonic Relevance Score) = RRF × G(d) × B(d,q)
   433	     * G(d) = Gravity: tier weight + log₂(1 + inDegree)
   434	     * B(d,q) = Blast Radius resonance: 1.5x if file imports the epicenter
   435	     *
   436	     * Untracked files (Dark Matter / Python / Go) gracefully degrade to baseline RRF scoring (Gravity = 1.0).
   437	     *
   438	     * @author Jherson Eddie Tintaya Holguin (Ruso-0)
   439	     */
   440	    async search(query: string, limit: number = 10): Promise<SearchResult[]> {
   441	        // 1. Fetch deep pool (5x limit) to overcome Semantic Dilution (ensure dependents make the cut)
   442	        const fetchLimit = limit * 5;
   443	        let rawResults: import("./database.js").HybridSearchResult[];
   444	
   445	        if (!this.config.enableEmbeddings) {
   446	            await this.initialize();
   447	            rawResults = this.db.searchKeywordOnly(query, fetchLimit);
   448	        } else {
   449	            await this.initializeEmbedder();
   450	            const { embedding } = await this.embedder.embed(query);
   451	            rawResults = this.db.searchHybrid(embedding, query, fetchLimit);
   452	        }
   453	
   454	        if (rawResults.length === 0) return [];
   455	
   456	        // 2. Load Topological Graph (cached after first call)
   457	        const graph = await this.getDependencyGraph();
   458	        const projectRoot = this.getProjectRoot();
   459	
   460	        // 3. Identify Epicenters (Top 3 pure semantic matches — where the definition lives)
   461	        const epicenters = new Set<string>();
   462	        for (const r of rawResults.slice(0, 3)) {
   463	            epicenters.add(path.relative(projectRoot, r.path).replace(/\\/g, "/"));
   464	        }
   465	
   466	        // 4. Apply Tectonic Relevance Scoring (TRS)
   467	        const scoredResults: SearchResult[] = rawResults.map(r => {
   468	            const relPath = path.relative(projectRoot, r.path).replace(/\\/g, "/");
   469	
   470	            // Orphan detection: files outside the dependency graph degrade gracefully
   471	            const isTracked = graph.inDegree.has(relPath);
   472	            const tier = graph.tiers.get(relPath);
   473	            const inDegree = graph.inDegree.get(relPath) || 0;
   474	
   475	            // Blast Radius detection: does this file import an epicenter?
   476	            let isBlastRadius = false;
   477	            for (const epi of epicenters) {
   478	                if (epi !== relPath && graph.importedBy.get(epi)?.has(relPath)) {
   479	                    isBlastRadius = true;
   480	                    break;
   481	                }
   482	            }
   483	
   484	            // Software Physics Math:
   485	            // Mass = PageRank Tier Weight (Core: 0.5, Logic: 0.2) + Logarithmic Gravity
   486	            let gravity = 1.0;
   487	            if (isTracked && tier) {
   488	                const wTier = tier === "core" ? 0.5 : (tier === "logic" ? 0.2 : 0.0);
   489	                // Damping factor prevents massive hubs from overwhelming semantic relevance
   490	                gravity = 1.0 + wTier + (0.15 * Math.log2(1 + inDegree));
   491	            }
   492	
   493	            // Blast Boost: +50% relevance if it's a dependent of an epicenter
   494	            const blastBoost = isBlastRadius ? 1.5 : 1.0;
   495	
   496	            // TRS = RRF * Gravity * BlastBoost
   497	            const trs = r.rrf_score * gravity * blastBoost;
   498	
   499	            // Top dependents for Sensory Injection (computed here, displayed in router.ts)
   500	            let topDependents: string[] = [];
   501	            if (epicenters.has(relPath) && inDegree > 0) {
   502	                const deps = graph.importedBy.get(relPath);
   503	                if (deps) {
   504	                    topDependents = Array.from(deps)
   505	                        .sort((a, b) => (graph.inDegree.get(b) || 0) - (graph.inDegree.get(a) || 0))
   506	                        .slice(0, 3)
   507	                        .map(d => path.basename(d));
   508	                }
   509	            }
   510	
   511	            return {
   512	                path: r.path,
   513	                shorthand: r.shorthand, // PURE DATA — never mutated by T-RAG
   514	                rawCode: r.raw_code,
   515	                nodeType: r.node_type,
   516	                startLine: r.start_line,
   517	                endLine: r.end_line,
   518	                score: trs,
   519	                topology: {
   520	                    tier: tier || "orphan",
   521	                    inDegree,
   522	                    isEpicenter: epicenters.has(relPath),
   523	                    isBlastRadius,
   524	                    dependents: topDependents,
   525	                },
   526	            };
   527	        });
   528	
   529	        // 5. Sort by TRS and return top K
   530	        scoredResults.sort((a, b) => b.score - a.score);
   531	        return scoredResults.slice(0, limit);
   532	    }

   559	
   560	    /** Compress a file using the advanced LLMLingua-2-inspired pipeline. */
   561	    async compressFileAdvanced(
   562	        filePath: string,
   563	        level: CompressionLevel = "medium",
   564	        preloadedContent?: string,
   565	    ): Promise<AdvancedCompressionResult> {
   566	        await this.initialize();
   567	
   568	        // Validate size ALWAYS - even with preloaded content (prevents bypass)
   569	        let sizeInBytes: number;
   570	        if (preloadedContent !== undefined) {
   571	            sizeInBytes = Buffer.byteLength(preloadedContent, "utf8");
   572	        } else {
   573	            try {
   574	                sizeInBytes = fs.statSync(filePath).size;
   575	            } catch {
   576	                throw new Error(`Cannot access file: ${filePath}`);
   577	            }
   578	        }
   579	        const filterResult = shouldProcess(filePath, sizeInBytes);
   580	        if (!filterResult.process) {
   581	            throw new Error(`File skipped: ${filterResult.reason}`);
   582	        }
   583	
   584	        const content = preloadedContent ?? readSource(filePath);
   585	        const result = await this.advancedCompressor.compress(filePath, content, level);
   586	
   587	        // Track session savings
   588	        const ext = path.extname(filePath).toLowerCase() || ".unknown";
   589	        this.sessionSavings.totalTokensSaved += result.tokensSaved;

    70	 * Jina embeddings output L2-normalized vectors (magnitude = 1),
    71	 * so cosine_similarity = dot_product (no sqrt/division needed).

    74	function fastSimilarity(a: Float32Array, b: Float32Array): number {
    75	    let dot = 0;
    76	    for (let i = 0; i < a.length; i++) {
    77	        dot += a[i] * b[i];
    78	    }
    79	    return dot;

   138	     * Serialize the vector index to a binary buffer.
   139	     * Format: [count:u32] + ([rowid:u32][vec:f32×dim])×count
   140	     * Note: rowid uses UInt32 - max 4,294,967,295. Sufficient for practical
   141	     * codebases (would require billions of INSERT/DELETE cycles to overflow).
   142	     */
   143	    serialize(): Buffer {
   144	        const entries = Array.from(this.vectors.entries());
   145	        const header = Buffer.alloc(4);
   146	        header.writeUInt32LE(entries.length);
   147	
   148	        const chunks: Buffer[] = [header];
   149	        for (const [rowid, vec] of entries) {
   150	            const idBuf = Buffer.alloc(4);
   151	            idBuf.writeUInt32LE(rowid);
   152	            chunks.push(idBuf);
   153	            // AUDIT FIX: Force deep copy to isolate SQLite writes from WASM memory
   154	            chunks.push(Buffer.from(new Uint8Array(vec.buffer, vec.byteOffset, vec.byteLength)));

   160	    static deserialize(buf: Buffer, dim: number): VectorIndex {
   161	        const index = new VectorIndex();
   162	        if (buf.length < 4) return index;
   163	
   164	        const count = buf.readUInt32LE(0);
   165	        let offset = 4;
   166	        const vecBytes = dim * 4;
   167	
   168	        for (let i = 0; i < count; i++) {
   169	            if (offset + 4 + vecBytes > buf.length) break;
   170	            const rowid = buf.readUInt32LE(offset);
   171	            offset += 4;
   172	            // Always deep-copy. Zero-copy via Float32Array(buf.buffer, offset, dim)
   173	            // creates a VIEW that shares memory with the source buffer.
   174	            // If sql.js WASM reuses that memory region, vectors corrupt silently.
   175	            const vec = new Float32Array(dim);
   176	            new Uint8Array(vec.buffer).set(buf.subarray(offset, offset + vecBytes));
   177	            index.vectors.set(rowid, vec);
   178	            offset += vecBytes;
   179	        }

   335	        let totalLen = 0;
   336	        for (const terms of this.docTerms.values()) {
   337	            totalLen += terms.length;
   338	        }
   339	        this.avgDocLen = totalLen / this.docCount;
   340	    }
   341	
   342	    /**
   343	     * BM25 search with bigram phrase boosting.
   344	     * Code-tuned parameters: k1 = 1.8, b = 0.35
   345	     * Multi-word queries get a 0.3 weight bigram boost.
   346	     */
   347	    search(
   348	        queryText: string,
   349	        limit: number
   350	    ): Array<{ rowid: number; score: number }> {
   351	        const queryTerms = this.tokenize(queryText);
   352	        if (queryTerms.length === 0) return [];
   353	
   354	        const k1 = 1.8;
   355	        const b = 0.35;
   356	        const scores = new Map<number, number>();
   357	
   358	        for (const term of queryTerms) {
   359	            const docMap = this.invertedIndex.get(term);
   360	            if (!docMap) continue;
   361	
   362	            // IDF = log((N - df + 0.5) / (df + 0.5) + 1)
   363	            const df = docMap.size;
   364	            const idf = Math.log(
   365	                (this.docCount - df + 0.5) / (df + 0.5) + 1
   366	            );
   367	
   368	            // TF read directly from inverted index - O(1)
   369	            for (const [rowid, tf] of docMap) {
   370	                const docLen = this.docTerms.get(rowid)!.length;
   371	
   372	                // BM25 formula
   373	                const tfNorm =
   374	                    (tf * (k1 + 1)) /
   375	                    (tf + k1 * (1 - b + b * (docLen / (this.avgDocLen || 1))));
   376	                const score = idf * tfNorm;
   377	
   378	                scores.set(rowid, (scores.get(rowid) || 0) + score);
   379	            }
   380	        }
   381	
   382	        // Bigram phrase boost for multi-word queries
   383	        if (queryTerms.length >= 2) {
   384	            const bigramWeight = 0.3;
   385	            for (let i = 0; i < queryTerms.length - 1; i++) {
   386	                const bigram = queryTerms[i] + "_" + queryTerms[i + 1];
   387	                const docs = this.bigramIndex.get(bigram);
   388	                if (!docs) continue;
   389	                for (const rowid of docs) {
   390	                    const existing = scores.get(rowid) || 0;
   391	                    scores.set(rowid, existing + bigramWeight);
   392	                }
   393	            }
   394	        }
   395	
   396	        return Array.from(scores.entries())
   397	            .map(([rowid, score]) => ({ rowid, score }))
   398	            .sort((a, b) => b.score - a.score)
   399	            .slice(0, limit);
   400	    }
   401	}

   751	    /** Apply path-based weighting: boost src/, penalize tests/node_modules/. */
   752	    private getPathBoost(filePath: string): number {
   753	        const normalized = filePath.replace(/\\/g, "/").toLowerCase();
   754	        if (normalized.includes("/node_modules/")) return 0.3;
   755	        if (normalized.includes("/dist/") || normalized.includes("/build/")) return 0.5;
   756	        if (normalized.includes("/test") || normalized.includes("/__test")) return 0.7;
   757	        if (normalized.includes("/src/")) return 1.2;
   758	        if (normalized.includes("/lib/") || normalized.includes("/core/")) return 1.1;
   759	        return 1.0;
   760	    }

   764	    /**
   765	     * Batch-fetch paths for an array of chunk IDs. Single SQL query.
   766	     * Used by RRF fusion to apply path boosting without N+1 queries.
   767	     */
   768	    private fetchPathsBatch(ids: number[]): Map<number, string> {
   769	        const result = new Map<number, string>();
   770	        if (ids.length === 0) return result;
   771	        const placeholders = ids.map(() => "?").join(",");
   772	        const stmt = this.db.prepare(
   773	            `SELECT id, path FROM chunks WHERE id IN (${placeholders})`,
   774	        );
   775	        try {
   776	            stmt.bind(ids);
   777	            while (stmt.step()) {
   778	                const row = stmt.getAsObject() as { id: number; path: string };
   779	                result.set(row.id, row.path);
   780	            }
   781	        } finally {
   782	            stmt.free();
   783	        }
   784	        return result;
   785	    }

   787	    /**
   788	     * Batch-fetch full chunk data for an array of chunk IDs. Single SQL query.
   789	     * Used by all search methods to hydrate final results without N+1 queries.
   790	     */
   791	    private fetchChunksBatch(ids: number[]): Map<number, ChunkRecord> {
   792	        const result = new Map<number, ChunkRecord>();
   793	        if (ids.length === 0) return result;
   794	        const placeholders = ids.map(() => "?").join(",");
   795	        const stmt = this.db.prepare(
   796	            `SELECT id, path, shorthand, raw_code, node_type, start_line, end_line, start_index, end_index, symbol_name
   797	             FROM chunks WHERE id IN (${placeholders})`,
   798	        );
   799	        try {
   800	            stmt.bind(ids);
   801	            while (stmt.step()) {
   802	                const row = stmt.getAsObject() as Record<string, number | string>;
   803	                result.set(row.id as number, {
   804	                    id: row.id as number,
   805	                    path: row.path as string,
   806	                    shorthand: row.shorthand as string,
   807	                    raw_code: row.raw_code as string,
   808	                    node_type: row.node_type as string,
   809	                    start_line: row.start_line as number,
   810	                    end_line: row.end_line as number,
   811	                    start_index: (row.start_index as number) ?? 0,
   812	                    end_index: (row.end_index as number) ?? 0,
   813	                    symbol_name: (row.symbol_name as string) ?? "",
   814	                });
   815	            }
   816	        } finally {
   817	            stmt.free();
   818	        }
   819	        return result;
   820	    }

   822	    // ─── Search Operations ───────────────────────────────────────
   823	
   824	    /**
   825	     * Hybrid search using Reciprocal Rank Fusion (RRF).
   826	     * Combines:
   827	     *   - Vector similarity (semantic, cosine distance)
   828	     *   - BM25 keyword matching (in pure JS inverted index)
   829	     *
   830	     * RRF formula: score = Σ 1/(k + rank_i) where k=10
   831	     */
   832	    searchHybrid(
   833	        queryEmbedding: Float32Array,
   834	        queryText: string,
   835	        limit: number = 10
   836	    ): HybridSearchResult[] {
   837	        // 1. Vector search - top 60 by cosine similarity
   838	        const vecResults = this.vecIndex.search(queryEmbedding, 60);
   839	        const vecRanks = new Map<number, number>();
   840	        vecResults.forEach((r, i) => vecRanks.set(r.rowid, i + 1));
   841	
   842	        // 2. BM25 keyword search - top 60 by term relevance
   843	        const kwResults = this.kwIndex.search(queryText, 60);
   844	        const kwRanks = new Map<number, number>();
   845	        kwResults.forEach((r, i) => kwRanks.set(r.rowid, i + 1));
   846	
   847	        // 3. RRF fusion with path boosting (batch query)
   848	        const allIds = new Set([...vecRanks.keys(), ...kwRanks.keys()]);
   849	        const pathMap = this.fetchPathsBatch([...allIds]);
   850	        const scored: Array<{ id: number; rrf: number }> = [];
   851	
   852	        for (const id of allIds) {
   853	            const vecRank = vecRanks.get(id);
   854	            const kwRank = kwRanks.get(id);
   855	            let rrf =
   856	                (vecRank ? 1.0 / (10 + vecRank) : 0) +
   857	                (kwRank ? 1.0 / (10 + kwRank) : 0);
   858	
   859	            const filePath = pathMap.get(id);
   860	            if (filePath) {
   861	                rrf *= this.getPathBoost(filePath);
   862	            }
   863	
   864	            scored.push({ id, rrf });
   865	        }
   866	
   867	        scored.sort((a, b) => b.rrf - a.rrf);
   868	        const topIds = scored.slice(0, limit);
   869	
   870	        // 4. Fetch full chunk data (batch query)
   871	        const chunkMap = this.fetchChunksBatch(topIds.map(t => t.id));
   872	        const results: HybridSearchResult[] = [];
   873	        for (const { id, rrf } of topIds) {
   874	            const row = chunkMap.get(id);
   875	            if (row) {
   876	                results.push({
   877	                    id: row.id, path: row.path, shorthand: row.shorthand,
   878	                    raw_code: row.raw_code, node_type: row.node_type,
   879	                    start_line: row.start_line, end_line: row.end_line,
   880	                    start_index: row.start_index, end_index: row.end_index,
   881	                    symbol_name: row.symbol_name,
   882	                    rrf_score: rrf,
   883	                });
   884	            }
   885	        }
   886	
   887	        return results;
   888	    }
   889	
   890	    /**
   891	     * Keyword-only search using BM25 (for Lite mode - no embeddings needed).
   892	     * Uses the in-memory KeywordIndex with path boosting.
   893	     */
   894	    searchKeywordOnly(
   895	        queryText: string,
   896	        limit: number = 10,
   897	    ): HybridSearchResult[] {
   898	        const kwResults = this.kwIndex.search(queryText, limit * 2);
   899	        if (kwResults.length === 0) return [];
   900	
   901	        const chunkMap = this.fetchChunksBatch(kwResults.map(r => r.rowid));
   902	        const results: HybridSearchResult[] = [];
   903	
   904	        for (const { rowid, score } of kwResults) {
   905	            const row = chunkMap.get(rowid);
   906	            if (row) {
   907	                const boostedScore = score * this.getPathBoost(row.path);
   908	                results.push({
   909	                    id: row.id, path: row.path, shorthand: row.shorthand,
   910	                    raw_code: row.raw_code, node_type: row.node_type,
   911	                    start_line: row.start_line, end_line: row.end_line,
   912	                    start_index: row.start_index, end_index: row.end_index,
   913	                    symbol_name: row.symbol_name,
   914	                    rrf_score: boostedScore,
   915	                });
   916	            }
   917	        }
   918	
   919	        results.sort((a, b) => b.rrf_score - a.rrf_score);
   920	        return results.slice(0, limit);
   921	    }
   922	
   923	    /**
   924	     * BM25-powered fast resolution for import-anchored auto-context.
   925	     * Searches "symbol pathHint" together to defeat homonyms.
   926	     * Enforces a 150ms hard timeout to prevent event loop blocking.
   927	     */
   928	    resolveImportSignatures(
   929	        deps: Array<{ symbol: string; pathHint: string }>,
   930	        maxTimeMs: number = 150,
   931	    ): Array<{ raw: string; path: string }> {
   932	        if (!this._ready || deps.length === 0) return [];
   933	
   934	        const start = performance.now();
   935	        const results: Array<{ raw: string; path: string }> = [];
   936	        const seenSymbols = new Set<string>();
   937	
   938	        for (const dep of deps) {
   939	            if (seenSymbols.has(dep.symbol)) continue;
   940	            seenSymbols.add(dep.symbol);
   941	            if (performance.now() - start > maxTimeMs) break;
   942	
   943	            // BM25 with two terms: symbol + path hint defeats homonyms
   944	            const cleanHint = dep.pathHint.replace(/['"%_]/g, " ").trim();
   945	            const queryText = cleanHint
   946	                ? `${dep.symbol} ${cleanHint}`
   947	                : dep.symbol;
   948	
   949	            const hits = this.searchKeywordOnly(queryText, 3);
   950	
   951	            if (hits.length > 0) {
   952	                // Final validation: symbol must appear textually in the shorthand
   953	                // Uses safe boundaries (not \b) to handle $store etc.
   954	                const safeSym = escapeRegExp(dep.symbol);
   955	                const exactRegex = new RegExp(
   956	                    `(^|[^a-zA-Z0-9_$])${safeSym}(?=[^a-zA-Z0-9_$]|$)`,
   957	                );
   958	
   959	                for (const hit of hits) {
   960	                    if (exactRegex.test(hit.shorthand)) {
   961	                        results.push({ raw: hit.shorthand, path: hit.path });
   962	                        break;
   963	                    }
   964	                }
   965	            }
   966	        }
   967	
   968	        return results;
   969	    }

   971	    searchVector(
   972	        queryEmbedding: Float32Array,
   973	        limit: number = 10
   974	    ): HybridSearchResult[] {
   975	        const vecResults = this.vecIndex.search(queryEmbedding, limit);
   976	        if (vecResults.length === 0) return [];
   977	
   978	        const chunkMap = this.fetchChunksBatch(vecResults.map(r => r.rowid));
   979	        const results: HybridSearchResult[] = [];
   980	
   981	        for (const { rowid, distance } of vecResults) {
   982	            const row = chunkMap.get(rowid);
   983	            if (row) {
   984	                results.push({
   985	                    id: row.id, path: row.path, shorthand: row.shorthand,
   986	                    raw_code: row.raw_code, node_type: row.node_type,
   987	                    start_line: row.start_line, end_line: row.end_line,
   988	                    start_index: row.start_index, end_index: row.end_index,
   989	                    symbol_name: row.symbol_name,
   990	                    rrf_score: 1 - distance,
   991	                });
   992	            }
   993	        }
   994	
   995	        return results;
   996	    }

  1013	    getUsageStats(since?: string): {
  1014	        total_input: number;
  1015	        total_output: number;
  1016	        total_saved: number;
  1017	        tool_calls: number;
  1018	    } {
  1019	        const whereClause = since ? "WHERE timestamp >= ?" : "";
  1020	        const params = since ? [since] : [];
  1021	
  1022	        const stmt = this.db.prepare(
  1023	            `SELECT
  1024	        COALESCE(SUM(input_tokens), 0)  AS total_input,
  1025	        COALESCE(SUM(output_tokens), 0) AS total_output,
  1026	        COALESCE(SUM(saved_tokens), 0)  AS total_saved,
  1027	        COUNT(*)                         AS tool_calls
  1028	      FROM usage_log ${whereClause}`
  1029	        );
  1030	
  1031	        try {
  1032	            if (params.length > 0) stmt.bind(params);
  1033	
  1034	            let result = { total_input: 0, total_output: 0, total_saved: 0, tool_calls: 0 };
  1035	            if (stmt.step()) {
  1036	                const row = stmt.getAsObject() as Record<string, number>;
  1037	                result = {
  1038	                    total_input: row.total_input ?? 0,
  1039	                    total_output: row.total_output ?? 0,
  1040	                    total_saved: row.total_saved ?? 0,
  1041	                    tool_calls: row.tool_calls ?? 0,
  1042	                };
  1043	            }
  1044	            return result;
  1045	        } finally {
  1046	            stmt.free();
  1047	        }
  1048	    }

  1051	     * Find the heaviest files by total raw code size.
  1052	     * Zero disk I/O - queries indexed data in SQLite.

  1054	    getTopHeavyFiles(limit: number = 5): Array<{ path: string; estimated_tokens: number }> {
  1055	        if (!this._ready) return [];
  1056	        const stmt = this.db.prepare(`
  1057	            SELECT path, SUM(LENGTH(raw_code)) as total_chars
  1058	            FROM chunks
  1059	            GROUP BY path
  1060	            ORDER BY total_chars DESC
  1061	            LIMIT ?
  1062	        `);
  1063	        try {
  1064	            stmt.bind([limit]);
  1065	
  1066	            const results: Array<{ path: string; estimated_tokens: number }> = [];
  1067	            while (stmt.step()) {
  1068	                const row = stmt.getAsObject() as { path: string; total_chars: number };
  1069	                results.push({
  1070	                    path: row.path,
  1071	                    estimated_tokens: Math.ceil(row.total_chars / 3.5),
  1072	                });
  1073	            }
  1074	            return results;
  1075	        } finally {

  1077	        }

  1127	    /**
  1128	     * Find all files whose raw code contains the given symbol name.
  1129	     * Returns distinct file paths. Used by prepare_refactor for 100% coverage.
  1130	     *
  1131	     * Uses the in-memory rawIdentsByFile index (built during insertChunk and
  1132	     * rebuildRawIdentIndex at startup). Substring match on identifiers — same
  1133	     * semantics as the previous LIKE %term% scan but O(unique_idents) instead
  1134	     * of O(total_raw_code_bytes).
  1135	     */
  1136	    searchRawCode(symbolName: string): string[] {
  1137	        if (!this._ready) return [];
  1138	        const results: string[] = [];
  1139	        for (const [filePath, idents] of this.rawIdentsByFile) {
  1140	            // PATCH-2: O(1) Set lookup instead of O(N) iteration with .includes().
  1141	            // Exact match eliminates false positives (e.g. "id" no longer matches "width").
  1142	            if (idents.has(symbolName)) { results.push(filePath); }
  1143	        }
  1144	        return results;
  1145	    }
