   211	    // ─── File Indexing ─────────────────────────────────────────────

   213	    /**
   214	     * Index a single file: parse AST → generate embeddings → store chunks.
   215	     * Skips files whose SHA-256 hash hasn't changed (Merkle diffing).
   216	     *
   217	     * In Lite mode, stores chunks without embeddings (keyword search only).
   218	     * In Pro mode, generates embeddings for hybrid search.
   219	     */
   220	    async indexFile(filePath: string): Promise<ParseResult | null> {
   221	        if (this.config.enableEmbeddings) {
   222	            await this.initializeEmbedder();
   223	        } else {
   224	            await this.initialize();
   225	        }
   226	
   227	        // FIX 7: Check file size and extension before processing
   228	        let stat: fs.Stats;
   229	        try {
   230	            stat = fs.statSync(filePath);
   231	        } catch {
   232	            return null;
   233	        }
   234	
   235	        const filterResult = shouldProcess(filePath, stat.size);
   236	        if (!filterResult.process) {
   237	            return null;
   238	        }
   239	
   240	        // Read file content (BOM-safe for tree-sitter)
   241	        let content: string;
   242	        try {
   243	            content = readSource(filePath);
   244	        } catch {
   245	            return null; // File may have been deleted
   246	        }
   247	
   248	        // Check extension support - unsupported files get plaintext fallback
   249	        const ext = path.extname(filePath).toLowerCase();
   250	        if (!this.parser.isSupported(ext)) {
   251	            // Plaintext fallback: index entire file as one chunk for BM25 keyword search.
   252	            // No AST features (no validation, no structural compression, no semantic edit)
   253	            // but search WILL find content in this file.
   254	            if (!this.db.fileNeedsUpdate(filePath, content)) return null;
   255	
   256	            const lineCount = content.split("\n").length;
   257	            // Use content as shorthand so BM25 keyword index can search it
   258	            const shorthand = `[file] ${path.basename(filePath)} (${lineCount} lines)\n${content}`;
   259	
   260	            this.db.clearChunks(filePath);
   261	
   262	            if (this.config.enableEmbeddings) {
   263	                await this.initializeEmbedder();
   264	                const { embedding } = await this.embedder.embed(content.slice(0, 1000));
   265	                this.db.insertChunk(filePath, shorthand, content, "file", 1, lineCount, embedding);
   266	            } else {
   267	                this.db.insertChunk(filePath, shorthand, content, "file", 1, lineCount, new Float32Array(0));
   268	            }
   269	
   270	            const hash = this.db.hashContent(content);
   271	            this.db.upsertFile(filePath, hash);
   272	
   273	            return {
   274	                filePath,
   275	                chunks: [{ shorthand, rawCode: content, nodeType: "file", startLine: 1, endLine: lineCount, startIndex: 0, endIndex: content.length, symbolName: "" }],
   276	                totalLines: lineCount,
   277	                language: ext.slice(1),
   278	            };
   279	        }
   280	
   281	        // Merkle diffing: skip unchanged files
   282	        if (!this.db.fileNeedsUpdate(filePath, content)) return null;
   283	
   284	        // Parse AST
   285	        const result = await this.parser.parse(filePath, content);
   286	        if (result.chunks.length === 0) return result;
   287	
   288	        // Clear old chunks for this file
   289	        this.db.clearChunks(filePath);
   290	
   291	        // Generate embeddings (or null placeholder) and store in batch
   292	        const chunkData: Array<{
   293	            path: string;
   294	            shorthand: string;
   295	            rawCode: string;
   296	            nodeType: string;
   297	            startLine: number;
   298	            endLine: number;
   299	            embedding: Float32Array;
   300	            startIndex: number;
   301	            endIndex: number;
   302	            symbolName: string;
   303	        }> = [];
   304	
   305	        for (const chunk of result.chunks) {
   306	            let embedding: Float32Array;
   307	            if (this.config.enableEmbeddings) {
   308	                ({ embedding } = await this.embedder.embed(chunk.shorthand));
   309	            } else {
   310	                // Lite mode: empty embedding placeholder
   311	                embedding = new Float32Array(0);
   312	            }
   313	            chunkData.push({
   314	                path: filePath,
   315	                shorthand: chunk.shorthand,
   316	                rawCode: chunk.rawCode,
   317	                nodeType: chunk.nodeType,
   318	                startLine: chunk.startLine,
   319	                endLine: chunk.endLine,
   320	                embedding,
   321	                startIndex: chunk.startIndex,
   322	                endIndex: chunk.endIndex,
   323	                symbolName: chunk.symbolName,
   324	            });
   325	        }
   326	
   327	        // Batch insert with transaction
   328	        this.db.insertChunksBatch(chunkData);
   329	
   330	        // Update file hash
   331	        const hash = this.db.hashContent(content);
   332	        this.db.upsertFile(filePath, hash);
   333	
   334	        return result;
   335	    }

   337	    /**
   338	     * Index an entire directory recursively.
   339	     * Walks all supported files and indexes them in sequence.
   340	     */
   341	    async indexDirectory(dirPath: string): Promise<{
   342	        indexed: number;
   343	        skipped: number;
   344	        errors: number;
   345	    }> {
   346	        if (this.config.enableEmbeddings) {
   347	            await this.initializeEmbedder();
   348	        } else {
   349	            await this.initialize();
   350	        }
   351	
   352	        let indexed = 0;
   353	        let skipped = 0;
   354	        let errors = 0;
   355	
   356	        const files = this.walkDirectory(dirPath);
   357	        let processedCount = 0;
   358	
   359	        for (const file of files) {
   360	            try {
   361	                const result = await this.indexFile(file);
   362	                if (result && result.chunks.length > 0) {
   363	                    indexed++;
   364	                } else {
   365	                    skipped++;
   366	                }
   367	            } catch (err) {
   368	                logger.error(
   369	                    `Error indexing ${file}: ${(err as Error).message}`
   370	                );
   371	                errors++;
   372	            }
   373	
   374	            // FIX 3: Yield event loop every 100 files to avoid blocking
   375	            processedCount++;
   376	            if (processedCount % 100 === 0) {
   377	                await new Promise<void>(resolve => setImmediate(resolve));
   378	            }
   379	        }
   380	
   381	        // Persist to disk after batch indexing
   382	        this.db.save();
   383	
   384	        return { indexed, skipped, errors };
   385	    }

   387	    /** Recursively walk a directory and return all supported files. */
   388	    private walkDirectory(dirPath: string): string[] {
   389	        const files: string[] = [];
   390	        const isIgnored = picomatch(this.config.ignorePaths);
   391	
   392	        const walk = (dir: string) => {
   393	            let entries: fs.Dirent[];
   394	            try {
   395	                entries = fs.readdirSync(dir, { withFileTypes: true });
   396	            } catch {
   397	                return;
   398	            }
   399	
   400	            for (const entry of entries) {
   401	                const fullPath = path.join(dir, entry.name);
   402	                const relativePath = path.relative(dirPath, fullPath).replace(/\\/g, "/");
   403	
   404	                if (entry.isDirectory()) {
   405	                    // Check directory name and relative path against glob patterns
   406	                    if (!isIgnored(entry.name) && !isIgnored(relativePath) && !isIgnored(relativePath + "/")) {
   407	                        walk(fullPath);
   408	                    }
   409	                    continue;
   410	                }
   411	
   412	                // Check file against ignore patterns, then check extension
   413	                if (isIgnored(relativePath) || isIgnored(entry.name)) continue;
   414	                const ext = path.extname(entry.name).toLowerCase();
   415	                if (this.config.extensions.includes(ext)) {
   416	                    files.push(fullPath);
   417	                }
   418	            }
   419	        };
   420	
   421	        walk(dirPath);
   422	        return files;
   423	    }

   427	    /**
   428	     * T-RAG (Topology-Aware RAG) - Blast Radius-Aware Search.
   429	     * Re-ranks standard semantic/keyword search results using the project's dependency graph.
   430	     * Injects PageRank (mass), InDegree (gravity), and Blast Radius awareness directly into the LLM's context.
   431	     *
   432	     * TRS (Tectonic Relevance Score) = RRF × G(d) × B(d,q)
   433	     * G(d) = Gravity: tier weight + log₂(1 + inDegree)
   434	     * B(d,q) = Blast Radius resonance: 1.5x if file imports the epicenter
   435	     *
   436	     * Untracked files (Dark Matter / Python / Go) gracefully degrade to baseline RRF scoring (Gravity = 1.0).
   437	     *
   438	     * @author Jherson Eddie Tintaya Holguin (Ruso-0)
   439	     */
   440	    async search(query: string, limit: number = 10): Promise<SearchResult[]> {
   441	        // 1. Fetch deep pool (5x limit) to overcome Semantic Dilution (ensure dependents make the cut)
   442	        const fetchLimit = limit * 5;
   443	        let rawResults: import("./database.js").HybridSearchResult[];
   444	
   445	        if (!this.config.enableEmbeddings) {
   446	            await this.initialize();
   447	            rawResults = this.db.searchKeywordOnly(query, fetchLimit);
   448	        } else {
   449	            await this.initializeEmbedder();
   450	            const { embedding } = await this.embedder.embed(query);
   451	            rawResults = this.db.searchHybrid(embedding, query, fetchLimit);
   452	        }
   453	
   454	        if (rawResults.length === 0) return [];
   455	
   456	        // 2. Load Topological Graph (cached after first call)
   457	        const graph = await this.getDependencyGraph();
   458	        const projectRoot = this.getProjectRoot();
   459	
   460	        // 3. Identify Epicenters (Top 3 pure semantic matches — where the definition lives)
   461	        const epicenters = new Set<string>();
   462	        for (const r of rawResults.slice(0, 3)) {
   463	            epicenters.add(path.relative(projectRoot, r.path).replace(/\\/g, "/"));
   464	        }
   465	
   466	        // 4. Apply Tectonic Relevance Scoring (TRS)
   467	        const scoredResults: SearchResult[] = rawResults.map(r => {
   468	            const relPath = path.relative(projectRoot, r.path).replace(/\\/g, "/");
   469	
   470	            // Orphan detection: files outside the dependency graph degrade gracefully
   471	            const isTracked = graph.inDegree.has(relPath);
   472	            const tier = graph.tiers.get(relPath);
   473	            const inDegree = graph.inDegree.get(relPath) || 0;
   474	
   475	            // Blast Radius detection: does this file import an epicenter?
   476	            let isBlastRadius = false;
   477	            for (const epi of epicenters) {
   478	                if (epi !== relPath && graph.importedBy.get(epi)?.has(relPath)) {
   479	                    isBlastRadius = true;
   480	                    break;
   481	                }
   482	            }
   483	
   484	            // Software Physics Math:
   485	            // Mass = PageRank Tier Weight (Core: 0.5, Logic: 0.2) + Logarithmic Gravity
   486	            let gravity = 1.0;
   487	            if (isTracked && tier) {
   488	                const wTier = tier === "core" ? 0.5 : (tier === "logic" ? 0.2 : 0.0);
   489	                // Damping factor prevents massive hubs from overwhelming semantic relevance
   490	                gravity = 1.0 + wTier + (0.15 * Math.log2(1 + inDegree));
   491	            }
   492	
   493	            // Blast Boost: +50% relevance if it's a dependent of an epicenter
   494	            const blastBoost = isBlastRadius ? 1.5 : 1.0;
   495	
   496	            // TRS = RRF * Gravity * BlastBoost
   497	            const trs = r.rrf_score * gravity * blastBoost;
   498	
   499	            // Top dependents for Sensory Injection (computed here, displayed in router.ts)
   500	            let topDependents: string[] = [];
   501	            if (epicenters.has(relPath) && inDegree > 0) {
   502	                const deps = graph.importedBy.get(relPath);
   503	                if (deps) {
   504	                    topDependents = Array.from(deps)
   505	                        .sort((a, b) => (graph.inDegree.get(b) || 0) - (graph.inDegree.get(a) || 0))
   506	                        .slice(0, 3)
   507	                        .map(d => path.basename(d));
   508	                }
   509	            }
   510	
   511	            return {
   512	                path: r.path,
   513	                shorthand: r.shorthand, // PURE DATA — never mutated by T-RAG
   514	                rawCode: r.raw_code,
   515	                nodeType: r.node_type,
   516	                startLine: r.start_line,
   517	                endLine: r.end_line,
   518	                score: trs,
   519	                topology: {
   520	                    tier: tier || "orphan",
   521	                    inDegree,
   522	                    isEpicenter: epicenters.has(relPath),
   523	                    isBlastRadius,
   524	                    dependents: topDependents,
   525	                },
   526	            };
   527	        });
   528	
   529	        // 5. Sort by TRS and return top K
   530	        scoredResults.sort((a, b) => b.score - a.score);
   531	        return scoredResults.slice(0, limit);
   532	    }

   559	
   560	    /** Compress a file using the advanced LLMLingua-2-inspired pipeline. */
   561	    async compressFileAdvanced(
   562	        filePath: string,
   563	        level: CompressionLevel = "medium",
   564	        preloadedContent?: string,
   565	    ): Promise<AdvancedCompressionResult> {
   566	        await this.initialize();
   567	
   568	        // Validate size ALWAYS - even with preloaded content (prevents bypass)
   569	        let sizeInBytes: number;
   570	        if (preloadedContent !== undefined) {
   571	            sizeInBytes = Buffer.byteLength(preloadedContent, "utf8");
   572	        } else {
   573	            try {
   574	                sizeInBytes = fs.statSync(filePath).size;
   575	            } catch {
   576	                throw new Error(`Cannot access file: ${filePath}`);
   577	            }
   578	        }
   579	        const filterResult = shouldProcess(filePath, sizeInBytes);
   580	        if (!filterResult.process) {
   581	            throw new Error(`File skipped: ${filterResult.reason}`);
   582	        }
   583	
   584	        const content = preloadedContent ?? readSource(filePath);
   585	        const result = await this.advancedCompressor.compress(filePath, content, level);
   586	
   587	        // Track session savings
   588	        const ext = path.extname(filePath).toLowerCase() || ".unknown";
   589	        this.sessionSavings.totalTokensSaved += result.tokensSaved;

   770	        if (this.isIndexing) return;

   772	        let madeChanges = false;
   773	
   774	        try {
   775	            while (this.indexingQueue.size > 0) {
   776	                const [filePath] = this.indexingQueue;
   777	                this.indexingQueue.delete(filePath);
   778

   781	                    if (res) madeChanges = true;

   786	                }
   787	            }
   788	        } finally {
   789	            if (madeChanges) {
   790	                this.db.save(); // Persist SQLite to disk after watcher changes
   791	            }
   792	            this.isIndexing = false; // Always reset to prevent deadlocks
   793	        }

    70	 * Jina embeddings output L2-normalized vectors (magnitude = 1),

    74	function fastSimilarity(a: Float32Array, b: Float32Array): number {
    75	    let dot = 0;
    76	    for (let i = 0; i < a.length; i++) {
    77	        dot += a[i] * b[i];
    78	    }
    79	    return dot;
