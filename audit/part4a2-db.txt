   155	        }
   156	        this._dirty = false;
   157	        return Buffer.concat(chunks);
   158	    }
   159	
   160	    static deserialize(buf: Buffer, dim: number): VectorIndex {
   161	        const index = new VectorIndex();
   162	        if (buf.length < 4) return index;
   163	
   164	        const count = buf.readUInt32LE(0);
   165	        let offset = 4;
   166	        const vecBytes = dim * 4;
   167	
   168	        for (let i = 0; i < count; i++) {
   169	            if (offset + 4 + vecBytes > buf.length) break;
   170	            const rowid = buf.readUInt32LE(offset);
   171	            offset += 4;
   172	            // Always deep-copy. Zero-copy via Float32Array(buf.buffer, offset, dim)
   173	            // creates a VIEW that shares memory with the source buffer.
   174	            // If sql.js WASM reuses that memory region, vectors corrupt silently.
   175	            const vec = new Float32Array(dim);
   176	            new Uint8Array(vec.buffer).set(buf.subarray(offset, offset + vecBytes));
   177	            index.vectors.set(rowid, vec);
   178	            offset += vecBytes;
   179	        }
   180	        return index;
   181	    }
   182	}

   186	// ─── In-Memory Keyword Index ─────────────────────────────────────────
   187	
   188	/**
   189	 * Pure JavaScript inverted index for BM25-style keyword search.
   190	 * Replaces FTS5 entirely - no native extensions needed.
   191	 *
   192	 * Tokenization: lowercases, splits on non-alphanumeric chars,
   193	 * filters stopwords, applies basic stemming (suffix removal).
   194	 */
   195	class KeywordIndex {
   196	    /** Map from term → Map<rowid, TF> - unified inverted index + term frequency */
   197	    private invertedIndex = new Map<string, Map<number, number>>();
   198	    /** Map from bigram → Set of document rowids (for phrase search) */
   199	    private bigramIndex = new Map<string, Set<number>>();
   200	    /** Map from rowid → tokenized terms (for delete and avgDocLen) */
   201	    private docTerms = new Map<number, string[]>();
   202	    /** Total number of documents */
   203	    private docCount = 0;
   204	    /** Average document length in terms */
   205	    private avgDocLen = 0;
   206	
   207	    private static STOPWORDS = new Set([
   208	        "a", "an", "the", "is", "are", "was", "were", "be", "been",
   209	        "being", "have", "has", "had", "do", "does", "did", "will",
   210	        "would", "could", "should", "may", "might", "shall", "can",
   211	        "to", "of", "in", "for", "on", "with", "at", "by", "from",
   212	        "as", "into", "through", "during", "before", "after", "above",
   213	        "below", "and", "but", "or", "not", "no", "if", "then",
   214	        "else", "this", "that", "it", "its", "new", "old",
   215	    ]);
   216	
   217	    /** Tokenize text into normalized terms with code-aware splitting. */
   218	    private tokenize(text: string): string[] {
   219	        // FIX 5: Apply code-aware tokenizer before stemming
   220	        const rawTokens = text
   221	            .replace(/[^a-zA-Z0-9_.]/g, " ")
   222	            .split(/\s+/)
   223	            .filter((t) => t.length > 1);
   224	
   225	        const allTerms: string[] = [];
   226	        for (const raw of rawTokens) {
   227	            // Code-aware tokenization: split identifiers
   228	            const subTokens = codeTokenize(raw);
   229	            if (subTokens.length > 0) {
   230	                for (const sub of subTokens) {
   231	                    if (sub.length > 1 && !KeywordIndex.STOPWORDS.has(sub)) {
   232	                        allTerms.push(sub);
   233	                    }
   234	                }
   235	            } else {
   236	                const lower = raw.toLowerCase();
   237	                if (!KeywordIndex.STOPWORDS.has(lower)) {
   238	                    allTerms.push(this.stem(lower));
   239	                }
   240	            }
   241	        }
   242	        return allTerms;
   243	    }
   244	
   245	    /** Stem cache — deterministic function, no reason to recompute. */
   246	    private stemCache = new Map<string, string>();
   247	
   248	    private stem(word: string): string {
   249	        if (word.length <= 2) return word;
   250	        let result = this.stemCache.get(word);
   251	        if (result === undefined) {
   252	            result = PorterStemmer.stem(word);
   253	            this.stemCache.set(word, result);
   254	        }
   255	        return result;
   256	    }
   257	
   258	    /** Add a document to the index. */
   259	    insert(rowid: number, text: string): void {
   260	        const terms = this.tokenize(text);
   261	        this.docTerms.set(rowid, terms);
   262	
   263	        // Compute local TF
   264	        const tfMap = new Map<string, number>();
   265	        for (const term of terms) {
   266	            tfMap.set(term, (tfMap.get(term) || 0) + 1);
   267	        }
   268	
   269	        // Store TF directly in inverted index for O(1) lookup
   270	        for (const [term, tf] of tfMap) {
   271	            let docMap = this.invertedIndex.get(term);
   272	            if (!docMap) {
   273	                docMap = new Map<number, number>();
   274	                this.invertedIndex.set(term, docMap);
   275	            }
   276	            docMap.set(rowid, tf);
   277	        }
   278	
   279	        // Generate bigrams for phrase search
   280	        for (let i = 0; i < terms.length - 1; i++) {
   281	            const bigram = terms[i] + "_" + terms[i + 1];
   282	            if (!this.bigramIndex.has(bigram)) {
   283	                this.bigramIndex.set(bigram, new Set());
   284	            }
   285	            this.bigramIndex.get(bigram)!.add(rowid);
   286	        }
   287	
   288	        this.docCount++;
   289	        this.updateAvgDocLen();
   290	    }
   291	
   292	    /** Remove a document from the index. */
   293	    delete(rowid: number): void {
   294	        const terms = this.docTerms.get(rowid);
   295	        if (!terms) return;
   296	
   297	        for (const term of terms) {
   298	            const docMap = this.invertedIndex.get(term);
   299	            if (docMap) {
   300	                docMap.delete(rowid);
   301	                if (docMap.size === 0) {
   302	                    this.invertedIndex.delete(term);
   303	                }
   304	            }
   305	        }
   306	
   307	        // Clean up bigram entries
   308	        for (let i = 0; i < terms.length - 1; i++) {
   309	            const bigram = terms[i] + "_" + terms[i + 1];
   310	            const docs = this.bigramIndex.get(bigram);
   311	            if (docs) {
   312	                docs.delete(rowid);
   313	                if (docs.size === 0) {
   314	                    this.bigramIndex.delete(bigram);
   315	                }
   316	            }
   317	        }
   318	
   319	        this.docTerms.delete(rowid);
   320	        this.docCount = Math.max(0, this.docCount - 1);
   321	        this.updateAvgDocLen();
   322	    }
   323	
   324	    deleteBulk(rowids: number[]): void {
   325	        for (const id of rowids) {
   326	            this.delete(id);
   327	        }
   328	    }
   329	
   330	    private updateAvgDocLen(): void {
   331	        if (this.docCount === 0) {
   332	            this.avgDocLen = 0;
   333	            return;
   334	        }
   335	        let totalLen = 0;
   336	        for (const terms of this.docTerms.values()) {
   337	            totalLen += terms.length;
   338	        }
   339	        this.avgDocLen = totalLen / this.docCount;
   340	    }
   341	
   342	    /**
   343	     * BM25 search with bigram phrase boosting.
   344	     * Code-tuned parameters: k1 = 1.8, b = 0.35
   345	     * Multi-word queries get a 0.3 weight bigram boost.
   346	     */
   347	    search(
   348	        queryText: string,
   349	        limit: number
   350	    ): Array<{ rowid: number; score: number }> {
   351	        const queryTerms = this.tokenize(queryText);
   352	        if (queryTerms.length === 0) return [];
   353	
   354	        const k1 = 1.8;
   355	        const b = 0.35;
   356	        const scores = new Map<number, number>();
   357	
   358	        for (const term of queryTerms) {
   359	            const docMap = this.invertedIndex.get(term);
   360	            if (!docMap) continue;
   361	
   362	            // IDF = log((N - df + 0.5) / (df + 0.5) + 1)
   363	            const df = docMap.size;
   364	            const idf = Math.log(
   365	                (this.docCount - df + 0.5) / (df + 0.5) + 1
   366	            );
   367	
   368	            // TF read directly from inverted index - O(1)
   369	            for (const [rowid, tf] of docMap) {
   370	                const docLen = this.docTerms.get(rowid)!.length;
   371	
   372	                // BM25 formula
   373	                const tfNorm =
   374	                    (tf * (k1 + 1)) /
   375	                    (tf + k1 * (1 - b + b * (docLen / (this.avgDocLen || 1))));
   376	                const score = idf * tfNorm;
   377	
   378	                scores.set(rowid, (scores.get(rowid) || 0) + score);
   379	            }
   380	        }
   381	
   382	        // Bigram phrase boost for multi-word queries
   383	        if (queryTerms.length >= 2) {
   384	            const bigramWeight = 0.3;
   385	            for (let i = 0; i < queryTerms.length - 1; i++) {
   386	                const bigram = queryTerms[i] + "_" + queryTerms[i + 1];
   387	                const docs = this.bigramIndex.get(bigram);
   388	                if (!docs) continue;
   389	                for (const rowid of docs) {
   390	                    const existing = scores.get(rowid) || 0;
   391	                    scores.set(rowid, existing + bigramWeight);
   392	                }
   393	            }
   394	        }
   395	
   396	        return Array.from(scores.entries())
   397	            .map(([rowid, score]) => ({ rowid, score }))
   398	            .sort((a, b) => b.score - a.score)
   399	            .slice(0, limit);
   400	    }
   401	}

   744	            this.db.run("ROLLBACK");

   749	    // ─── Path Boosting ────────────────────────────────────────────

   751	    /** Apply path-based weighting: boost src/, penalize tests/node_modules/. */
   752	    private getPathBoost(filePath: string): number {
   753	        const normalized = filePath.replace(/\\/g, "/").toLowerCase();
   754	        if (normalized.includes("/node_modules/")) return 0.3;
   755	        if (normalized.includes("/dist/") || normalized.includes("/build/")) return 0.5;
   756	        if (normalized.includes("/test") || normalized.includes("/__test")) return 0.7;
   757	        if (normalized.includes("/src/")) return 1.2;
   758	        if (normalized.includes("/lib/") || normalized.includes("/core/")) return 1.1;
   759	        return 1.0;
   760	    }

   764	    /**
   765	     * Batch-fetch paths for an array of chunk IDs. Single SQL query.
   766	     * Used by RRF fusion to apply path boosting without N+1 queries.
   767	     */
   768	    private fetchPathsBatch(ids: number[]): Map<number, string> {
   769	        const result = new Map<number, string>();
   770	        if (ids.length === 0) return result;
   771	        const placeholders = ids.map(() => "?").join(",");
   772	        const stmt = this.db.prepare(
   773	            `SELECT id, path FROM chunks WHERE id IN (${placeholders})`,
   774	        );
   775	        try {
   776	            stmt.bind(ids);
   777	            while (stmt.step()) {
   778	                const row = stmt.getAsObject() as { id: number; path: string };
   779	                result.set(row.id, row.path);
   780	            }
   781	        } finally {
   782	            stmt.free();
   783	        }
   784	        return result;
   785	    }

   787	    /**
   788	     * Batch-fetch full chunk data for an array of chunk IDs. Single SQL query.
   789	     * Used by all search methods to hydrate final results without N+1 queries.
   790	     */
   791	    private fetchChunksBatch(ids: number[]): Map<number, ChunkRecord> {
   792	        const result = new Map<number, ChunkRecord>();
   793	        if (ids.length === 0) return result;
   794	        const placeholders = ids.map(() => "?").join(",");
   795	        const stmt = this.db.prepare(
   796	            `SELECT id, path, shorthand, raw_code, node_type, start_line, end_line, start_index, end_index, symbol_name
   797	             FROM chunks WHERE id IN (${placeholders})`,
   798	        );
   799	        try {
   800	            stmt.bind(ids);
   801	            while (stmt.step()) {
   802	                const row = stmt.getAsObject() as Record<string, number | string>;
   803	                result.set(row.id as number, {
   804	                    id: row.id as number,
   805	                    path: row.path as string,
   806	                    shorthand: row.shorthand as string,
   807	                    raw_code: row.raw_code as string,
   808	                    node_type: row.node_type as string,
   809	                    start_line: row.start_line as number,
   810	                    end_line: row.end_line as number,
   811	                    start_index: (row.start_index as number) ?? 0,
   812	                    end_index: (row.end_index as number) ?? 0,
   813	                    symbol_name: (row.symbol_name as string) ?? "",
   814	                });
   815	            }
   816	        } finally {
   817	            stmt.free();
   818	        }
   819	        return result;
   820	    }

   822	    // ─── Search Operations ───────────────────────────────────────
   823	
   824	    /**
   825	     * Hybrid search using Reciprocal Rank Fusion (RRF).
   826	     * Combines:
   827	     *   - Vector similarity (semantic, cosine distance)
   828	     *   - BM25 keyword matching (in pure JS inverted index)
   829	     *
   830	     * RRF formula: score = Σ 1/(k + rank_i) where k=10
   831	     */
   832	    searchHybrid(
   833	        queryEmbedding: Float32Array,
   834	        queryText: string,
   835	        limit: number = 10
   836	    ): HybridSearchResult[] {
   837	        // 1. Vector search - top 60 by cosine similarity
   838	        const vecResults = this.vecIndex.search(queryEmbedding, 60);
   839	        const vecRanks = new Map<number, number>();
   840	        vecResults.forEach((r, i) => vecRanks.set(r.rowid, i + 1));
   841	
   842	        // 2. BM25 keyword search - top 60 by term relevance
   843	        const kwResults = this.kwIndex.search(queryText, 60);
   844	        const kwRanks = new Map<number, number>();
   845	        kwResults.forEach((r, i) => kwRanks.set(r.rowid, i + 1));
   846	
   847	        // 3. RRF fusion with path boosting (batch query)
   848	        const allIds = new Set([...vecRanks.keys(), ...kwRanks.keys()]);
   849	        const pathMap = this.fetchPathsBatch([...allIds]);
   850	        const scored: Array<{ id: number; rrf: number }> = [];
   851	
   852	        for (const id of allIds) {
   853	            const vecRank = vecRanks.get(id);
   854	            const kwRank = kwRanks.get(id);
   855	            let rrf =
   856	                (vecRank ? 1.0 / (10 + vecRank) : 0) +
   857	                (kwRank ? 1.0 / (10 + kwRank) : 0);
   858	
   859	            const filePath = pathMap.get(id);
   860	            if (filePath) {
   861	                rrf *= this.getPathBoost(filePath);
   862	            }
   863	
   864	            scored.push({ id, rrf });
   865	        }
   866	
   867	        scored.sort((a, b) => b.rrf - a.rrf);
   868	        const topIds = scored.slice(0, limit);
   869	
   870	        // 4. Fetch full chunk data (batch query)
   871	        const chunkMap = this.fetchChunksBatch(topIds.map(t => t.id));
   872	        const results: HybridSearchResult[] = [];
   873	        for (const { id, rrf } of topIds) {
   874	            const row = chunkMap.get(id);
   875	            if (row) {
   876	                results.push({
   877	                    id: row.id, path: row.path, shorthand: row.shorthand,
   878	                    raw_code: row.raw_code, node_type: row.node_type,
   879	                    start_line: row.start_line, end_line: row.end_line,
   880	                    start_index: row.start_index, end_index: row.end_index,
   881	                    symbol_name: row.symbol_name,
   882	                    rrf_score: rrf,
   883	                });
   884	            }
   885	        }
   886	
   887	        return results;
   888	    }

   890	    /**
   891	     * Keyword-only search using BM25 (for Lite mode - no embeddings needed).
   892	     * Uses the in-memory KeywordIndex with path boosting.
   893	     */
   894	    searchKeywordOnly(
   895	        queryText: string,
   896	        limit: number = 10,
   897	    ): HybridSearchResult[] {
   898	        const kwResults = this.kwIndex.search(queryText, limit * 2);
   899	        if (kwResults.length === 0) return [];
   900	
   901	        const chunkMap = this.fetchChunksBatch(kwResults.map(r => r.rowid));
   902	        const results: HybridSearchResult[] = [];
   903	
   904	        for (const { rowid, score } of kwResults) {
   905	            const row = chunkMap.get(rowid);
   906	            if (row) {
   907	                const boostedScore = score * this.getPathBoost(row.path);
   908	                results.push({
   909	                    id: row.id, path: row.path, shorthand: row.shorthand,
   910	                    raw_code: row.raw_code, node_type: row.node_type,
   911	                    start_line: row.start_line, end_line: row.end_line,
   912	                    start_index: row.start_index, end_index: row.end_index,
   913	                    symbol_name: row.symbol_name,
   914	                    rrf_score: boostedScore,
   915	                });
   916	            }
   917	        }
   918	
   919	        results.sort((a, b) => b.rrf_score - a.rrf_score);
   920	        return results.slice(0, limit);
   921	    }

   923	    /**
   924	     * BM25-powered fast resolution for import-anchored auto-context.
   925	     * Searches "symbol pathHint" together to defeat homonyms.
   926	     * Enforces a 150ms hard timeout to prevent event loop blocking.
   927	     */
   928	    resolveImportSignatures(
   929	        deps: Array<{ symbol: string; pathHint: string }>,
   930	        maxTimeMs: number = 150,
   931	    ): Array<{ raw: string; path: string }> {
   932	        if (!this._ready || deps.length === 0) return [];
   933	
   934	        const start = performance.now();
   935	        const results: Array<{ raw: string; path: string }> = [];
   936	        const seenSymbols = new Set<string>();
   937	
   938	        for (const dep of deps) {
   939	            if (seenSymbols.has(dep.symbol)) continue;
   940	            seenSymbols.add(dep.symbol);
   941	            if (performance.now() - start > maxTimeMs) break;
   942	
   943	            // BM25 with two terms: symbol + path hint defeats homonyms
   944	            const cleanHint = dep.pathHint.replace(/['"%_]/g, " ").trim();
   945	            const queryText = cleanHint
   946	                ? `${dep.symbol} ${cleanHint}`
   947	                : dep.symbol;
   948	
   949	            const hits = this.searchKeywordOnly(queryText, 3);
   950	
   951	            if (hits.length > 0) {
   952	                // Final validation: symbol must appear textually in the shorthand
   953	                // Uses safe boundaries (not \b) to handle $store etc.
   954	                const safeSym = escapeRegExp(dep.symbol);
   955	                const exactRegex = new RegExp(
   956	                    `(^|[^a-zA-Z0-9_$])${safeSym}(?=[^a-zA-Z0-9_$]|$)`,
   957	                );
   958	
   959	                for (const hit of hits) {
   960	                    if (exactRegex.test(hit.shorthand)) {
   961	                        results.push({ raw: hit.shorthand, path: hit.path });
   962	                        break;
   963	                    }
   964	                }
   965	            }
   966	        }
   967	
   968	        return results;
   969	    }

   971	    searchVector(
   972	        queryEmbedding: Float32Array,
   973	        limit: number = 10
   974	    ): HybridSearchResult[] {
   975	        const vecResults = this.vecIndex.search(queryEmbedding, limit);
   976	        if (vecResults.length === 0) return [];
   977	
   978	        const chunkMap = this.fetchChunksBatch(vecResults.map(r => r.rowid));
   979	        const results: HybridSearchResult[] = [];
   980	
   981	        for (const { rowid, distance } of vecResults) {
   982	            const row = chunkMap.get(rowid);
   983	            if (row) {
   984	                results.push({
   985	                    id: row.id, path: row.path, shorthand: row.shorthand,
   986	                    raw_code: row.raw_code, node_type: row.node_type,
   987	                    start_line: row.start_line, end_line: row.end_line,
   988	                    start_index: row.start_index, end_index: row.end_index,
   989	                    symbol_name: row.symbol_name,
   990	                    rrf_score: 1 - distance,
   991	                });
   992	            }
   993	        }
   994	
   995	        return results;
   996	    }

  1127	    /**
  1128	     * Find all files whose raw code contains the given symbol name.
  1129	     * Returns distinct file paths. Used by prepare_refactor for 100% coverage.
  1130	     *
  1131	     * Uses the in-memory rawIdentsByFile index (built during insertChunk and
  1132	     * rebuildRawIdentIndex at startup). Substring match on identifiers — same
  1133	     * semantics as the previous LIKE %term% scan but O(unique_idents) instead
  1134	     * of O(total_raw_code_bytes).
  1135	     */
  1136	    searchRawCode(symbolName: string): string[] {
